122       return (this->rep_ && s.rep_)

155     ACE_INLINE ACE_SString::size_type
156     ACE_SString::find (const char *s, size_type pos) const
157     {
158       char *substr = this->rep_ + pos;
159       char *pointer = ACE_OS::strstr (substr, s);
160       if (pointer == nullptr)
161         return ACE_SString::npos;
162       else
163         return static_cast<ACE_SString::size_type> (pointer - this->rep_);
164     }

166     ACE_INLINE ACE_SString::size_type
167     ACE_SString::find (char c, size_type pos) const
168     {
169       char *substr = this->rep_ + pos;
170       char *pointer = ACE_OS::strchr (substr, c);
171       if (pointer == nullptr)
172         return ACE_SString::npos;
173       else
174         return static_cast<ACE_SString::size_type> (pointer - this->rep_);
175     }

191     ACE_INLINE ACE_SString::size_type
192     ACE_SString::rfind (char c, size_type pos) const
193     {
194       if (pos == ACE_SString::npos)
195         pos = this->len_;
196     
197       // Do not change to prefix operator!  Proper operation of this loop
198       // depends on postfix decrement behavior.
199       for (size_type i = pos; i-- != 0; )
200         if (this->rep_[i] == c)
201           return i;
202     
203       return ACE_SString::npos;
204     }

232     ACE_INLINE void
233     ACE_Auto_String_Free::reset (char* p)
234     {
235     #if defined (ACE_HAS_ALLOC_HOOKS)
236       ACE_Allocator::instance()->free (this->p_);
237     #else
238       ACE_OS::free (this->p_);
239     #endif /* ACE_HAS_ALLOC_HOOKS */
240       this->p_ = p;
241     }

244     ACE_Auto_String_Free::operator= (ACE_Auto_String_Free& rhs)
245     {
246       if (this != std::addressof(rhs))
247         {
248           this->reset (rhs.p_);
249           rhs.p_ = nullptr;
250         }
251       return *this;
252     }
