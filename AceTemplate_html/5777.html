42     /**
43      * @class ACE_TSS
44      *
45      * @brief Allows objects that are "physically" in thread specific
46      * storage (i.e., private to a thread) to be accessed as though
47      * they were "logically" global to a program.
48      *
49      * This class helps to maintain a separate copy of an object for each thread
50      * that needs access to it. All threads access a single instance of ACE_TSS
51      * to obtain a pointer to a thread-specific copy of a TYPE object. Using
52      * a pointer to TYPE in TSS instead of TYPE itself is useful because,
53      * in addition to avoiding copies on what may be a complex class, it allows
54      * assignment of objects to thread-specific data that have arbitrarily
55      * complex constructors.
56      *
57      * When the ACE_TSS object is destroyed, all threads's instances of the
58      * data are deleted.
59      *
60      * Modern compilers have no problem using a built-in type for @c TYPE.
61      * However, if you must use an older compiler that won't work with a built-in
62      * type, the ACE_TSS_Type_Adapter class template, below, can be used for
63      * adapting built-in types to work with ACE_TSS.
64      *
65      * @note Beware when creating static instances of this type
66      * (as with any other, btw). The unpredictable order of initialization
67      * across different platforms may cause a situation where one uses
68      * the instance before it is fully initialized. That's why typically
69      * instances of this type are dynamicaly allocated. On the stack it is
70      * typically allocated inside the ACE_Thread::svc() method which
71      * limits its lifetime appropriately.
72      */
73     template <class TYPE>
74     class ACE_TSS
75     {
76     public:
77       /**
78        * Default constructor. Can also initialize this ACE_TSS instance,
79        * readying it for use by the calling thread as well as all other
80        * threads in the process. If the constructor does not initialize this
81        * object, the first access to it will perform the initialization, which
82        * could possibly (under odd error conditions) fail.
83        *
84        * @param ts_obj   If non-zero, this object is initialized for use by
85        *                 all threads and @a ts_obj is used to set the
86        *                 thread-specific value for the calling thread. Other
87        *                 threads use the ts_object (TYPE *) method to set
88        *                 a specific value.
89        */
90       ACE_TSS (TYPE *ts_obj = nullptr);
91     
92       ACE_TSS (const ACE_TSS &) = delete;
93       ACE_TSS (ACE_TSS &&) = delete;
94       ACE_TSS &operator= (const ACE_TSS &) = delete;
95       ACE_TSS &operator= (ACE_TSS &&) = delete;
96     
97       /// Deregister this object from thread-specific storage administration.
98       /// Will cause all threads' copies of TYPE to be destroyed.
99       virtual ~ACE_TSS ();
100     
101       /**
102        * Set the thread-specific object for the calling thread.
103        * If this object has not been initialized yet, this method performs the
104        * initialization.
105        *
106        * @param new_ts_obj  The new value for the calling thread's copy of
107        *                    this object.
108        *
109        * @return  The previous value of the calling thread's copy of this
110        *          object; 0 if there was no previous value. This method also
111        *          returns 0 on errors. To tell the difference between an error
112        *          and a returned 0 pointer, it's recommended that one set errno
113        *          to 0 prior to calling ts_object() and check for a new errno
114        *          value if ts_object() returns 0.
115        */
116       TYPE *ts_object (TYPE *new_ts_obj);
117     
118       /** @name Accessors
119        *
120        * All accessors return a pointer to the calling thread's copy of the
121        * TYPE data. The pointer may be 0 on error conditions or if the calling
122        * thread's copy of the data has not yet been set. See specific method
123        * descriptions for complete details.
124        */
125       //@{
126       /**
127        * Get the thread-specific object for this object.
128        *
129        * @return  0 if the object has never been initialized, otherwise returns
130        *          the calling thread's copy of the data. The returned pointer
131        *          may be 0 under odd error conditions; check errno for further
132        *          information.
133        */
134       TYPE *ts_object () const;
135     
136       /**
137        * Use a "smart pointer" to get the thread-specific data associated
138        * with this object.
139        * If this ACE_TSS object hasn't been initialized, this method
140        * will initialize it as a side-affect. If the calling thread has not
141        * set a value, a default-constructed instance of TYPE is allocated and it
142        * becomes the thread's instance.
143        *
144        * @return  The calling thread's copy of the data. The returned pointer
145        *          may be 0 under odd error conditions; check errno for further
146        *          information.
147        */
148       TYPE *operator-> () const;
149     
150       /**
151        * Obtain a pointer to the calling thread's TYPE object.
152        * If this ACE_TSS object hasn't been initialized, this method
153        * will initialize it as a side-affect. If the calling thread has not
154        * set a value, a default-constructed instance of TYPE is allocated and it
155        * becomes the thread's instance.
156        *
157        * @return  The calling thread's copy of the data. The returned pointer
158        *          may be 0 under odd error conditions; check errno for further
159        *          information.
160        */
161       operator TYPE *() const;
162     
163       //@}
164     
165       /// Hook for construction parameters.
166       virtual TYPE *make_TSS_TYPE () const;
167     
168       // = Utility methods.
169     
170       /// Dump the state of an object.
171       void dump () const;
172     
173       /// Declare the dynamic allocation hooks.
174       ACE_ALLOC_HOOK_DECLARE;
175     
176     protected:
177       /// Actually implements the code that retrieves the object from
178       /// thread-specific storage.
179       TYPE *ts_get () const;
180     
181       /// Factors out common code for initializing TSS.  This must NOT be
182       /// called with the lock held...
183       int ts_init ();
184     
185     #if !(defined (ACE_HAS_THREADS) && (defined (ACE_HAS_THREAD_SPECIFIC_STORAGE) || defined (ACE_HAS_TSS_EMULATION)))
186       /// This implementation only works for non-threading systems...
187       TYPE *type_;
188     #else
189       /// Avoid race conditions during initialization.
190       ACE_Thread_Mutex keylock_;
191     
192       /// "First time in" flag.
193       volatile bool once_;
194     
195       /// Key for the thread-specific error data.
196       ACE_thread_key_t key_;
197     
198       /// "Destructor" that deletes internal TYPE * when thread exits.
199       static void cleanup (void *ptr);
200     
201       /// Obtains a plain value stored in the thread-specific storage.
202     # if defined (ACE_HAS_THR_C_DEST)
203       ACE_TSS_Adapter *ts_value () const;
204     # else
205       TYPE *ts_value () const;
206     # endif /* ACE_HAS_THR_C_DEST */
207     
208       /// Stores a new plain value in the thread-specific storage.
209     # if defined (ACE_HAS_THR_C_DEST)
210       int ts_value (ACE_TSS_Adapter *new_tss_adapter) const;
211     # else
212       int ts_value (TYPE *new_ts_obj) const;
213     # endif /* ACE_HAS_THR_C_DEST */
214     #endif /* defined (ACE_HAS_THREADS) && (defined (ACE_HAS_THREAD_SPECIFIC_STORAGE) || defined (ACE_HAS_TSS_EMULATION)) */
215     };

217     /**
218      * @class ACE_TSS_Type_Adapter
219      *
220      * @brief Adapter that allows built-in types to be used with ACE_TSS.
221      *
222      * Wraps a value of a built-in type, providing conversions to
223      * and from the type.  Example use with ACE_TSS:
224      * ACE_TSS<ACE_TSS_Type_Adapter<int> > i;
225      * *i = 37;
226      * ACE_OS::fprintf (stderr, "%d\n", *i);
227      * Unfortunately, though, some compilers have trouble with the
228      * implicit type conversions.  This seems to work better:
229      * ACE_TSS<ACE_TSS_Type_Adapter<int> > i;
230      * i->operator int & () = 37;
231      * ACE_OS::fprintf (stderr, "%d\n", i->operator int ());
232      */
233     template <class TYPE>
234     class ACE_TSS_Type_Adapter
235     {
236     public:
237       /// Constructor.  Inlined here so that it should _always_ be inlined.
238       ACE_TSS_Type_Adapter (const TYPE value = 0): value_ (value) {}
239     
240       /// TYPE conversion.  Inlined here so that it should _always_ be
241       /// inlined.
242       operator TYPE () const { return value_; }
243     
244       /// TYPE & conversion.  Inlined here so that it should _always_ be
245       /// inlined.
246       operator TYPE &() { return value_; }
247     
248     private:
249       /// The wrapped value.
250       TYPE value_;
251     };
