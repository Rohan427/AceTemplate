31       ACE_INLINE int
32       fstat (ACE_HANDLE handle, ACE_stat *stp)
33       {
34         ACE_OS_TRACE ("ACE_OS::fstat");
35     #if defined (ACE_WIN32)
36         BY_HANDLE_FILE_INFORMATION fdata;
37     
38         if (::GetFileInformationByHandle (handle, &fdata) == FALSE)
39           {
40             ACE_OS::set_errno_to_last_error ();
41             return -1;
42           }
43         if ((fdata.nFileSizeHigh != 0) && (sizeof (stp->st_size) < sizeof (ULONGLONG)))
44           {
45             errno = EINVAL; // return an error rather than incorrect values
46             return -1;
47           }
48     
49     #if defined (_FILE_OFFSET_BITS) && _FILE_OFFSET_BITS == 64
50         ULARGE_INTEGER ul;
51         ul.HighPart = fdata.nFileSizeHigh;
52         ul.LowPart = fdata.nFileSizeLow;
53         stp->st_size = ul.QuadPart;
54     #else
55         stp->st_size = fdata.nFileSizeLow;
56     #endif /* _FILE_OFFSET_BITS */
57         stp->st_atime = ACE_Time_Value (fdata.ftLastAccessTime).sec ();
58         stp->st_mtime = ACE_Time_Value (fdata.ftLastWriteTime).sec ();
59         stp->st_ctime = ACE_Time_Value (fdata.ftCreationTime).sec ();
60         stp->st_nlink = static_cast<short> (fdata.nNumberOfLinks);
61         stp->st_dev = stp->st_rdev = 0; // No equivalent conversion.
62         mode_t const BASE_MODE = S_IXOTH | S_IROTH;
63         mode_t const WRITE_MODE = (fdata.dwFileAttributes & FILE_ATTRIBUTE_READONLY) ? 0 : S_IWOTH;
64         mode_t const TYPE_MODE = (fdata.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) ? S_IFDIR : S_IFREG;
65         stp->st_mode = BASE_MODE | WRITE_MODE | TYPE_MODE;
66         return 0;
67     #elif defined (ACE_LACKS_FSTAT)
68         ACE_UNUSED_ARG (handle);
69         ACE_UNUSED_ARG (stp);
70         ACE_NOTSUP_RETURN (-1);
71     #elif defined (ACE_MQX)
72         return MQX_Filesystem::inst ().fstat (handle, stp);
73     #else
74         return ::fstat (handle, stp);
75     #endif /* !ACE_WIN32 */
76       }

78       // This function returns the number of bytes in the file referenced by
79       // FD.
80       ACE_INLINE ACE_OFF_T
81       filesize (ACE_HANDLE handle)
82       {
83         ACE_OS_TRACE ("ACE_OS::filesize");
84     #if defined (ACE_WIN32)
85     # if defined (_FILE_OFFSET_BITS) && _FILE_OFFSET_BITS == 64
86         LARGE_INTEGER size;
87         return
88           (::GetFileSizeEx (handle, &size)
89            ? size.QuadPart
90            : (ACE_OS::set_errno_to_last_error (), -1));
91     # else
92         DWORD const size = ::GetFileSize (handle, 0);
93         return
94           (size != INVALID_FILE_SIZE
95            ? static_cast<ACE_OFF_T> (size)
96            : (ACE_OS::set_errno_to_last_error (), -1));
97     # endif  /* _FILE_OFFSET_BITS == 64 */
98     #else /* !ACE_WIN32 */
99         ACE_stat sb;
100         return ACE_OS::fstat (handle, &sb) == -1 ?
101                         static_cast<ACE_OFF_T> (-1) : sb.st_size;
102     #endif
103       }

105       ACE_INLINE ACE_OFF_T
106       filesize (const ACE_TCHAR *filename)
107       {
108         ACE_OS_TRACE ("ACE_OS::filesize");
109     
110     #if defined (ACE_LACKS_STAT)
111         ACE_HANDLE const h = ACE_OS::open (filename, O_RDONLY);
112         if (h != ACE_INVALID_HANDLE)
113           {
114             ACE_OFF_T const size = ACE_OS::filesize (h);
115             ACE_OS::close (h);
116             return size;
117           }
118         else
119           return -1;
120     #else /* !ACE_LACKS_STAT */
121         ACE_stat sb;
122         return ACE_OS::stat (filename, &sb) == -1 ?
123                         static_cast<ACE_OFF_T> (-1) : sb.st_size;
124     #endif /* ACE_LACKS_STAT */
125       }

229       ACE_INLINE mode_t
230       umask (mode_t cmask)
231       {
232         ACE_OS_TRACE ("ACE_OS::umask");
233     # if defined (ACE_LACKS_UMASK)
234         ACE_UNUSED_ARG (cmask);
235         ACE_NOTSUP_RETURN ((mode_t)-1);
236     # elif defined (ACE_HAS_TR24731_2005_CRT)
237         int old_mode;
238         int const new_mode = static_cast<int> (cmask);
239         ACE_SECURECRTCALL (_umask_s (new_mode, &old_mode), mode_t, -1, old_mode);
240         return static_cast<mode_t> (old_mode);
241     # elif defined (ACE_WIN32) && !defined (__BORLANDC__)
242         return ::_umask (cmask);
243     # else
244         return ::umask (cmask); // This call shouldn't fail...
245     # endif /* ACE_LACKS_UMASK */
246       }
