26     ACE_INLINE ACE_HANDLE
27     ACE_OS::accept (ACE_HANDLE handle,
28                     struct sockaddr *addr,
29                     int *addrlen)
30     {
31       ACE_OS_TRACE ("ACE_OS::accept");
32       // On a non-blocking socket with no connections to accept, this
33       // system call will return EWOULDBLOCK or EAGAIN, depending on the
34       // platform.  UNIX 98 allows either errno, and they may be the same
35       // numeric value.  So to make life easier for upper ACE layers as
36       // well as application programmers, always change EAGAIN to
37       // EWOULDBLOCK.  Rather than hack the ACE_OSCALL_RETURN macro, it's
38       // handled explicitly here.  If the ACE_OSCALL macro ever changes,
39       // this function needs to be reviewed.  On Win32, the regular macros
40       // can be used, as this is not an issue.
41     
42     #if defined (ACE_LACKS_ACCEPT)
43       ACE_UNUSED_ARG (handle);
44       ACE_UNUSED_ARG (addr);
45       ACE_UNUSED_ARG (addrlen);
46       ACE_NOTSUP_RETURN (ACE_INVALID_HANDLE);
47     #elif defined (ACE_WIN32)
48       ACE_SOCKCALL_RETURN (::accept ((ACE_SOCKET) handle,
49                                      addr,
50                                      (ACE_SOCKET_LEN *) addrlen),
51                            ACE_HANDLE,
52                            ACE_INVALID_HANDLE);
53     #else
54     #  if defined (ACE_HAS_BROKEN_ACCEPT_ADDR)
55       // Apparently some platforms like VxWorks can't correctly deal with
56       // a NULL addr.
57     
58     #    if defined (ACE_HAS_IPV6)
59        sockaddr_in6 fake_addr;
60     #    else
61        sockaddr_in fake_addr;
62     #    endif /* ACE_HAS_IPV6 */
63        int fake_addrlen;
64     
65        if (addrlen == 0)
66          addrlen = &fake_addrlen;
67     
68        if (addr == 0)
69          {
70            addr = (sockaddr *) &fake_addr;
71            *addrlen = sizeof fake_addr;
72          }
73     #  endif /* ACE_HAS_BROKEN_ACCEPT_ADDR */
74       ACE_HANDLE ace_result = ::accept ((ACE_SOCKET) handle,
75                                         addr,
76                                         (ACE_SOCKET_LEN *) addrlen);
77     
78     # if !(defined (EAGAIN) && defined (EWOULDBLOCK) && EAGAIN == EWOULDBLOCK)
79       // Optimize this code out if we can detect that EAGAIN ==
80       // EWOULDBLOCK at compile time.  If we cannot detect equality at
81       // compile-time (e.g. if EAGAIN or EWOULDBLOCK are not preprocessor
82       // macros) perform the check at run-time.  The goal is to avoid two
83       // TSS accesses in the _REENTRANT case when EAGAIN == EWOULDBLOCK.
84       if (ace_result == ACE_INVALID_HANDLE
85     #  if !defined (EAGAIN) || !defined (EWOULDBLOCK)
86           && EAGAIN != EWOULDBLOCK
87     #  endif  /* !EAGAIN || !EWOULDBLOCK */
88           && errno == EAGAIN)
89         {
90           errno = EWOULDBLOCK;
91         }
92     # endif /* EAGAIN != EWOULDBLOCK*/
93     
94       return ace_result;
95     
96     #endif /* defined (ACE_WIN32) */
97     }

183     ACE_INLINE int
184     ACE_OS::getpeername (ACE_HANDLE handle, struct sockaddr *addr,
185                          int *addrlen)
186     {
187       ACE_OS_TRACE ("ACE_OS::getpeername");
188     
189     #if defined (ACE_LACKS_GETPEERNAME)
190       ACE_UNUSED_ARG (handle);
191       ACE_UNUSED_ARG (addr);
192       ACE_UNUSED_ARG (addrlen);
193       ACE_NOTSUP_RETURN (-1);
194     #elif defined (ACE_GETNAME_RETURNS_RANDOM_SIN_ZERO) \
195                && (ACE_GETNAME_RETURNS_RANDOM_SIN_ZERO == 1)
196       int result;
197       ACE_SOCKCALL (::getpeername ((ACE_SOCKET) handle,
198                                    addr,
199                                    (ACE_SOCKET_LEN *) addrlen),
200                    int,
201                     -1,
202                     result);
203     
204       // Some platforms, like older versions of the Linux kernel, do not
205       // initialize the sin_zero field since that field is generally only
206       // used for padding/alignment purposes.  On those platforms
207       // memcmp()-based comparisons of the sockaddr_in structure, such as
208       // the one in the ACE_INET_Addr equality operator, may fail due to
209       // random bytes in the sin_zero field even though that field is
210       // unused.  Prevent equality comparison of two different sockaddr_in
211       // instances that refer to the same socket from failing by
212       // explicitly initializing the sockaddr_in::sin_zero field to a
213       // consistent value, e.g. zero.
214       if (result != -1 && addr->sa_family == AF_INET)
215         {
216           ACE_OS::memset (reinterpret_cast<struct sockaddr_in *> (addr)->sin_zero,
217                           0,
218                           sizeof (reinterpret_cast<struct sockaddr_in *> (addr)->sin_zero));
219         }
220     
221       return result;
222     #else
223       ACE_SOCKCALL_RETURN (::getpeername ((ACE_SOCKET) handle,
224                                           addr,
225                                           (ACE_SOCKET_LEN *) addrlen),
226                            int,
227                            -1);
228     #endif /* ACE_GETNAME_RETURNS_RANDOM_SIN_ZERO */
229     }

498     ACE_INLINE ssize_t
499     ACE_OS::recvv (ACE_HANDLE handle,
500                    iovec *buffers,
501                    int n)
502     {
503     #if defined (ACE_HAS_WINSOCK2)
504     
505       DWORD bytes_received = 0;
506       int result = 1;
507     
508       // Winsock 2 has WSARecv and can do this directly, but Winsock 1 needs
509       // to do the recvs piece-by-piece.
510     
511     # if (ACE_HAS_WINSOCK2 != 0)
512       DWORD flags = 0;
513       result = ::WSARecv ((SOCKET) handle,
514                           (WSABUF *) buffers,
515                           n,
516                           &bytes_received,
517                           &flags,
518                           0,
519                           0);
520     # else
521       // Step through the buffers requested by caller; for each one, cycle
522       // through reads until it's filled or an error occurs.
523       for (int i = 0; i < n && result > 0; ++i)
524         {
525           char *chunkp = buffers[i].iov_base;     // Point to part of chunk being read
526           int chunklen = buffers[i].iov_len;    // Track how much to read to chunk
527           while (chunklen > 0 && result > 0)
528             {
529               result = ::recv ((SOCKET) handle, chunkp, chunklen, 0);
530               if (result > 0)
531                 {
532                   chunkp += result;
533                   chunklen -= result;
534                   bytes_received += result;
535                 }
536             }
537         }
538     # endif /* ACE_HAS_WINSOCK2 != 0 */
539     
540       if (result == SOCKET_ERROR)
541         {
542           ACE_OS::set_errno_to_wsa_last_error ();
543           return -1;
544         }
545       else
546         return (ssize_t) bytes_received;
547     #else
548       return ACE_OS::readv (handle, buffers, n);
549     #endif /* ACE_HAS_WINSOCK2 */
550     }

622     # if (defined (ACE_HAS_WINSOCK2) && (ACE_HAS_WINSOCK2 != 0))
623       DWORD bytes_sent = 0;
624       int const result = msg->msg_control
625         ? sendmsg_win32_i (handle, msg, flags, bytes_sent)
626         : ::WSASendTo ((SOCKET) handle,
627                        (WSABUF *) msg->msg_iov,
628                        msg->msg_iovlen,
629                        &bytes_sent,
630                        flags,
631                        msg->msg_name,
632                        msg->msg_namelen,
633                        0,
634                        0);
635     
636       if (result != 0)
637         {
638           ACE_OS::set_errno_to_wsa_last_error ();
639           return -1;
640         }
641       else
642         return (ssize_t) bytes_sent;
643     # elif defined (ACE_HAS_NONCONST_SENDMSG)
644       ACE_SOCKCALL_RETURN (::sendmsg (handle,
645                                       const_cast<struct msghdr *>(msg),
646                                       flags), ssize_t, -1);
647     # else
648       ACE_SOCKCALL_RETURN (::sendmsg (handle, msg, flags), ssize_t, -1);
649     # endif
650     #else
651       ACE_UNUSED_ARG (flags);
652       ACE_UNUSED_ARG (msg);
653       ACE_UNUSED_ARG (handle);
654     
655       ACE_NOTSUP_RETURN (-1);
656     #endif /* ACE_LACKS_SENDMSG */
657     }

711     ACE_INLINE ssize_t
712     ACE_OS::sendto (ACE_HANDLE handle,
713                     const iovec *buffers,
714                     int buffer_count,
715                     size_t &number_of_bytes_sent,
716                     int flags,
717                     const struct sockaddr *addr,
718                     int addrlen,
719                     ACE_OVERLAPPED *overlapped,
720                     ACE_OVERLAPPED_COMPLETION_FUNC func)
721     {
722       ACE_OS_TRACE ("ACE_OS::sendto");
723     #if defined (ACE_HAS_WINSOCK2) && (ACE_HAS_WINSOCK2 != 0)
724       DWORD bytes_sent = 0;
725       int result = ::WSASendTo ((SOCKET) handle,
726                                 (WSABUF*) buffers,
727                                 buffer_count,
728                                 &bytes_sent,
729                                 flags,
730                                 addr,
731                                 addrlen,
732                                 overlapped,
733                                 func);
734       if (result != 0) {
735         ACE_OS::set_errno_to_wsa_last_error ();
736       }
737       number_of_bytes_sent = static_cast<size_t> (bytes_sent);
738       return (ssize_t) result;
739     #else
740       ACE_UNUSED_ARG (overlapped);
741       ACE_UNUSED_ARG (func);
742     
743       number_of_bytes_sent = 0;
744     
745       ssize_t result = 0;
746     
747       for (int i = 0; i < buffer_count; ++i)
748         {
749            result = ACE_OS::sendto (handle,
750                                     reinterpret_cast<char *> (
751                                                      buffers[i].iov_base),
752                                     buffers[i].iov_len,
753                                     flags,
754                                     addr,
755                                     addrlen);
756            if (result == -1)
757              break;
758            number_of_bytes_sent += static_cast<size_t> (result);
759         }
760     
761       return result;
762     #endif /* defined (ACE_HAS_WINSOCK2) && (ACE_HAS_WINSOCK2 != 0) */
763     }

765     ACE_INLINE ssize_t
766     ACE_OS::sendv (ACE_HANDLE handle,
767                    const iovec *buffers,
768                    int n)
769     {
770     #if defined (ACE_HAS_WINSOCK2)
771       DWORD bytes_sent = 0;
772       ssize_t result = 0;
773     
774       // Winsock 2 has WSASend and can do this directly, but Winsock 1
775       // needs to do the sends one-by-one.
776     # if (ACE_HAS_WINSOCK2 != 0) && !defined (ACE_DONT_USE_WSASEND)
777       result = ::WSASend ((SOCKET) handle,
778                           (WSABUF *) buffers,
779                           n,
780                           &bytes_sent,
781                           0,
782                           0,
783                           0);
784       if (result == SOCKET_ERROR)
785         {
786           ACE_OS::set_errno_to_wsa_last_error ();
787           if ((errno != ENOBUFS) ||
788               (bytes_sent != 0))
789             {
790               return -1;
791             }
792           result = sendv_partial_i(handle, buffers, n);
793           if (result == SOCKET_ERROR)
794             {
795               ACE_OS::set_errno_to_wsa_last_error ();
796               return -1;
797             }
798           bytes_sent = static_cast<DWORD>(result);
799         }
800     # else
801       for (int i = 0; i < n; ++i)
802         {
803           result = ::send ((SOCKET) handle,
804                            buffers[i].iov_base,
805                            buffers[i].iov_len,
806                            0);
807     
808           if (result == SOCKET_ERROR)
809             {
810               // There is a subtle difference in behaviour depending on
811               // whether or not any data was sent.  If no data was sent,
812               // then always return -1.  Otherwise return bytes_sent.
813               // This gives the caller an opportunity to keep track of
814               // bytes that have already been sent.
815               if (bytes_sent > 0)
816                 break;
817               else
818                 {
819                   ACE_OS::set_errno_to_wsa_last_error ();
820                   return -1;
821                 }
822             }
823           else
824             {
825               // Gets ignored on error anyway
826               bytes_sent += result;
827     
828               // If the transfer isn't complete just drop out of the loop.
829               if (result < (int)buffers[i].iov_len)
830                 break;
831             }
832         }
833     # endif /* ACE_HAS_WINSOCK2 != 0 */
834     
835       return (ssize_t) bytes_sent;
836     
837     #elif defined (ACE_MQX)
838       ssize_t bytes_sent = 0;
839       for (int i = 0; i < n; ++i)
840         {
841           ssize_t result = ACE_OS::send (handle, buffers[i].iov_base,
842                                          buffers[i].iov_len, 0);
843     
844           if (result == -1)
845             {
846               // There is a subtle difference in behaviour depending on
847               // whether or not any data was sent.  If no data was sent,
848               // then always return -1.  Otherwise return bytes_sent.
849               // This gives the caller an opportunity to keep track of
850               // bytes that have already been sent.
851               if (bytes_sent > 0)
852                 break;
853               else
854                 {
855                   // errno should already be set from the ACE_OS::send call.
856                   return -1;
857                 }
858             }
859           else
860             {
861               // Gets ignored on error anyway
862               bytes_sent += result;
863     
864               // If the transfer isn't complete just drop out of the loop.
865               if (result < buffers[i].iov_len)
866                 break;
867             }
868         }
869     
870       return bytes_sent;
871     #elif defined (ACE_HAS_SOCK_BUF_SIZE_MAX)
872     
873       // Platform limits the maximum socket message size.  Pare down the
874       // iovec, if necessary, to obey the limit.
875       iovec local_iov[ACE_IOV_MAX];
876       long total = 0;
877       long new_total = 0;
878       for (int i = 0; i < n; i++)
879         {
880           local_iov[i].iov_base = buffers[i].iov_base;
881           local_iov[i].iov_len  = buffers[i].iov_len;
882     
883           new_total = total + buffers[i].iov_len;
884           if (new_total >= ACE_HAS_SOCK_BUF_SIZE_MAX_VALUE)
885             {
886               local_iov[i].iov_len = ACE_HAS_SOCK_BUF_SIZE_MAX_VALUE - total;
887               n = i+1;
888               break;
889             }
890           total = new_total;
891         }
892       return ACE_OS::writev (handle, local_iov, n);
893     #else
894       return ACE_OS::writev (handle, buffers, n);
895     #endif /* ACE_HAS_WINSOCK2 */
896     }
