292     #       if defined (ACE_HAS_TSS_EMULATION) && !defined (ACE_HAS_THREAD_SPECIFIC_STORAGE)
293     class ACE_INTEGRITY_TSS_Impl
294     {
295     private:
296       struct TaskInfo
297       {
298         ACE_hthread_t id {};
299         void **tss_base {};
300         bool valid {};
301       };
302     
303       // Using std::map caused a memory issue, so we're using an array.
304       // Downside is the number of threads is fixed.
305       // May try ACE's map.
306       typedef TaskInfo TSS_Table[ACE_MAX_NUM_THREADS];
307     
308     public:
309       // Return a reference to the TSS base address of a given thread.
310       // If there exists an entry for the thread, just return the reference.
311       // If not, set an entry for it and return the reference.
312       void **&get (ACE_hthread_t id)
313       {
314         LockGuard guard (lock_);
315         unsigned first_slot = ACE_MAX_NUM_THREADS;
316         for (unsigned i = 0; i < ACE_MAX_NUM_THREADS; ++i)
317           {
318             TaskInfo &ti = integrity_ts_storage_[i];
319             if (!ti.valid)
320               {
321                 // Use a slot from an already finished thread.
322                 if (first_slot == ACE_MAX_NUM_THREADS)
323                   first_slot = i;
324               }
325             else if (ti.id == id)
326               {
327                 // This is not the first call to this function from the same thread.
328                 // Earlier call has already insert this entry.
329                 // Just return now.
330                 return ti.tss_base;
331               }
332           }
333         // If we go here, the calling thread doesn't have an entry yet.
334         // Use the first available slot.
335     
336         // Should NOT create more than ACE_MAX_NUM_THREADS Tasks!
337         ACE_ASSERT (first_slot != ACE_MAX_NUM_THREADS);
338         if (first_slot == ACE_MAX_NUM_THREADS)
339           {
340             ACE_OS::exit (EXIT_FAILURE);
341           }
342     
343         TaskInfo &my_ti = integrity_ts_storage_[first_slot];
344         my_ti.id = id;
345         my_ti.valid = true;
346         return my_ti.tss_base;
347       }
348     
349       // Mark the entry for a thread invalid (e.g. the thread has finished or exited).
350       // Return true if success, false if there is no entry for the thread.
351       bool erase (ACE_hthread_t id)
352       {
353         LockGuard guard (lock_);
354         for (unsigned i = 0; i < ACE_MAX_NUM_THREADS; ++i)
355           {
356             TaskInfo &ti = integrity_ts_storage_[i];
357             if (ti.id == id)
358               {
359                 ti.valid = false;
360                 return true;
361               }
362           }
363         return false;
364       }
365     
366     private:
367       TSS_Table integrity_ts_storage_;
368       ACE_thread_mutex_t lock_;
369     };

533     // Recursive mutex support.
534     //
535     // There are two parts to this:
536     // 1. The mutex type itself. This is based on whether or not the
537     //    platform supports recursive mutexes natively or they're emulated.
538     // 2. Support for using the recursive mutex with a condition variable.
539     //    When a thread waits on a condition variable, it has to relinquish
540     //    the lock and wait atomically, then reacquire it after the condition
541     //    variable is signaled. In non-recursive mutexes, the platform
542     //    handles this automatically. But in recursive mutexes, especially
543     //    when emulated, the recursion count needs to be maintained across
544     //    the wait. Since another thread needs to be able to acquire the
545     //    lock, it needs to appear free, even if the waiting thread had done
546     //    multiple acquires. Thus, there's another structure to hold this
547     //    information, and is used with the recursive_mutex_cond_unlock()
548     //    and recursive_mutex_cond_relock() methods to maintain the expected
549     //    state when the wait finishes.
550     #   if defined (ACE_HAS_RECURSIVE_MUTEXES)
551     
552     ACE_BEGIN_VERSIONED_NAMESPACE_DECL
553     
554     typedef ACE_thread_mutex_t ACE_recursive_thread_mutex_t;
555     #     if defined (ACE_WIN32)
556     // Windows has recursive mutexes, but doesn't have condition variables,
557     // so there's no built-in support for this. Thus, the condition-related
558     // unlock/relock is augmented in ACE.
559     struct ACE_recursive_mutex_state
560     {
561       // On Windows the augmented processing is simply unlocking/relocking
562       // the recursive locks - the condition handles a single lock ok.
563       LONG relock_count_;
564     };
565     #     else
566     // No need for special handling; just need a type for method signatures.
567     typedef int ACE_recursive_mutex_state;
568     #     endif /* ACE_WIN32 */
