46     #if defined (ACE_LACKS_IFREQ)
47     struct  ifreq {
48     #define IFNAMSIZ        16
49             char    ifr_name[IFNAMSIZ];             /* if name, e.g. "en0" */
50             union {
51                     struct  sockaddr ifru_addr;
52                     struct  sockaddr ifru_dstaddr;
53                     struct  sockaddr ifru_broadaddr;
54                     short   ifru_flags;
55                     int     ifru_metric;
56                     int     ifru_mtu;
57                     int     ifru_phys;
58                     int     ifru_media;
59                     caddr_t ifru_data;
60                     int     (*ifru_tap)(struct ifnet *, struct ether_header *, struct mbuf *);
61             } ifr_ifru;
62     #define ifr_addr        ifr_ifru.ifru_addr      /* address */
63     #define ifr_dstaddr     ifr_ifru.ifru_dstaddr   /* other end of p-to-p link */
64     #define ifr_broadaddr   ifr_ifru.ifru_broadaddr /* broadcast address */
65     #define ifr_flags       ifr_ifru.ifru_flags     /* flags */
66     #define ifr_metric      ifr_ifru.ifru_metric    /* metric */
67     #define ifr_mtu         ifr_ifru.ifru_mtu       /* mtu */
68     #define ifr_phys        ifr_ifru.ifru_phys      /* physical wire */
69     #define ifr_media       ifr_ifru.ifru_media     /* physical media */
70     #define ifr_data        ifr_ifru.ifru_data      /* for use by interface */
71     #define ifr_tap         ifr_ifru.ifru_tap       /* tap function */
72     };
73     #endif /* ACE_LACKS_IFREQ */
74
