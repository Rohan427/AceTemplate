43     // If the user doesn't supply a ACE_Message_Queue pointer then we'll
44     // allocate one dynamically.  Otherwise, we'll use the one they give.
45     
46     template<ACE_SYNCH_DECL, class TIME_POLICY>
47     ACE_Task<ACE_SYNCH_USE, TIME_POLICY>::ACE_Task (ACE_Thread_Manager *thr_man,
48                                        ACE_Message_Queue<ACE_SYNCH_USE, TIME_POLICY> *mq)
49       : ACE_Task_Base (thr_man),
50         msg_queue_ (0),
51         delete_msg_queue_ (false),
52         mod_ (0),
53         next_ (0)
54     {
55       ACE_TRACE ("ACE_Task<ACE_SYNCH_USE, TIME_POLICY>::ACE_Task");
56     
57       if (mq == 0)
58         {
59           typedef ACE_Message_Queue<ACE_SYNCH_USE, TIME_POLICY> QUEUE_TYPE;
60           ACE_NEW (mq,
61                    QUEUE_TYPE);
62           this->delete_msg_queue_ = true;
63         }
64     
65       this->msg_queue_ = mq;
66     }
67     
68     template<ACE_SYNCH_DECL, class TIME_POLICY>
69     ACE_Task<ACE_SYNCH_USE, TIME_POLICY>::~ACE_Task ()
70     {
71       ACE_TRACE ("ACE_Task<ACE_SYNCH_USE, TIME_POLICY>::~ACE_Task");
72       if (this->delete_msg_queue_)
73         delete this->msg_queue_;
74     
75       // These assignments aren't strickly necessary but they help guard
76       // against odd race conditions...
77       this->delete_msg_queue_ = false;
78     }
