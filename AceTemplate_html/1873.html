77     /**
78      * @class ACE_Object_Manager
79      *
80      * @brief Manager for ACE library services and singleton cleanup.
81      *
82      * The ACE_Object_Manager manages cleanup of objects, typically
83      * singletons, at program termination.  In addition to managing
84      * the cleanup of the ACE library, it provides an interface for
85      * application to register objects to be cleaned up.
86      * This class also shuts down ACE library services, so that they
87      * can reclaim their storage, at program termination.  It works
88      * by creating a static instance whose destructor gets called
89      * along with those of all other static objects.  Hooks are
90      * provided for application code to register objects and arrays
91      * for cleanup, e.g., destruction.  The order of such cleanup
92      * calls is in the reverse order of registration, i.e., that
93      * last object/array to register gets cleaned up first.
94      * The ACE_Object_Manager API includes ACE_Managed_Object.  That
95      * class is contained in a separate file because it is a
96      * template class, and some compilers require that template and
97      * non-template class definitions appear in separate files.
98      * Please see ace/Managed_Object.h for a description of that
99      * part of the API.  In summary, ACE_Managed_Object provides two
100      * adapters, the ACE_Cleanup_Adapter and ACE_Managed_Object
101      * template classes for adapting objects of any type to be
102      * easily managed by the ACE_Object_Manager.  There are several
103      * mechanisms for adapting objects and arrays for cleanup at
104      * program termination, in roughly increasing order of ease-of-use:
105      * 1) Derive the object's class from ACE_Cleanup.
106      * 2) Allow the ACE_Object_Manager to both dynamically allocate
107      * and deallocate the object.
108      * 3) Provide an <ACE_CLEANUP_FUNC> cleanup hook for the object or
109      * array.
110      * 4) Allow the ACE_Object_Manager to both preallocate the object
111      * or array, either statically in global data or dynamically on
112      * the heap, when its singleton instance is construction.
113      *
114      * There are also several mechanisms for registering objects and
115      * arrays for cleanup.  In decreasing order of flexibility and
116      * complexity (with the exception of the last mechanism):
117      *
118      * 1) ACE_Object_Manager::at_exit (void *object,
119      * ACE_CLEANUP_FUNC cleanup_hook,
120      * void *param);
121      * can be used to register any object or array for any
122      * cleanup activity at program termination.
123      * 2) ACE_Object_Manager::at_exit (ACE_Cleanup *object,
124      * void *param = nullptr);
125      * can be used to register an ACE_Cleanup object
126      * for any cleanup activity at program termination.
127      * The final mechanism is not general purpose, but can only
128      * be used to allocate objects and arrays at program startup:
129      * 3) ACE_Managed_Object::get_preallocated_object
130      * (ACE_Object_Manager::Preallocated_Object id);
131      * and
132      * ACE_Managed_Object::get_preallocated_array
133      * (ACE_Object_Manager::Preallocated_Array id);
134      * can only be used to allocate objects at program startup,
135      * either in global data or on the heap (selected at compile
136      * time).  These are intended to replace static locks, etc.
137      * Instead of creating a static ACE_Object_Manager instance, one
138      * can alternatively be created on the stack of the main program
139      * thread.  It is created just after entry to ::main (int, char
140      * *[]), and before any existing code in that function is
141      * executed.  To enable this alternative, add #define
142      * ACE_HAS_NONSTATIC_OBJECT_MANAGER before including the platform
143      * specific config-* file in ace/config.h prior to
144      * building the ACE library and your applications.  This #define
145      * is enabled in some config files that are supplied with ACE.
146      *
147      * To ensure a static object manager is used, #undef
148      * ACE_HAS_NONSTATIC_OBJECT_MANAGER *after* including the platform
149      * specific config-* file.
150      * Note that the ACE_Object_Manager _must_ be created before
151      * any threads are spawned by the program.
152      * If ACE_HAS_NONSTATIC_OBJECT_MANAGER is not #defined, the ACE
153      * library creates a static, singleton ACE_Object_Manager instance.
154      * The instance is placed in global program data, and constructed
155      * via a static object constructor.  If ACE_HAS_NONSTATIC_OBJECT_MANAGER
156      * is #defined, the ACE_Object_Manager instance is created on the stack
157      * of the main program thread, as noted above.
158      *
159      * With ACE_HAS_NONSTATIC_OBJECT_MANAGER enabled, the ACE
160      * library has no static objects that require destruction.
161      * However, there are two drawbacks to using it:
162      * 1) main (int, char *[]) must be declared with arguments, even
163      * if they're not used.  All of ACE is converted to this, so
164      * just applications have to be concerned with it.
165      * 2) If there any static objects that depend on those that are
166      * cleaned up by the Object_Manager, they'll get cleaned up too
167      * late.  The ACE tests do not violate this requirement.
168      * However, applications may have trouble with it.
169      * NOTE on the use of <::exit> -- <::exit> does not destroy
170      * automatic objects.  Therefore, if
171      * ACE_HAS_NONSTATIC_OBJECT_MANAGER is enabled, the
172      * ACE_Object_Manager instance will *not* be destroyed if
173      * <::exit> is called!  However, <ACE_OS::exit> will properly
174      * destroy the ACE_Object_Manager.  It is highly recommended
175      * that <ACE_OS::exit> be used instead of <::exit>.
176      *
177      * However, <::exit> and <ACE_OS::exit> are tricky to use
178      * properly, especially in multithread programs.  It is much
179      * safer to throw an exception (or simulate that effect) that
180      * will be caught by <main> instead of calling exit.  Then,
181      * <main> can perform any necessary application-specific cleanup
182      * and return the status value.  In addition, it's usually best
183      * to avoid calling <::exit> and <ACE_OS::exit> from threads
184      * other than the main thread.  Thanks to Jeff Greif
185      * <jmg@trivida.com> for pointing out that <::exit> doesn't
186      * destroy automatic objects, and for developing the
187      * recommendations in this paragraph.
188      *
189      * Instead of creating a static ACE_Object_Manager, or letting
190      * ACE create it on the stack of <main> for you, another
191      * alternative is to #define
192      * ACE_DOESNT_INSTANTIATE_NONSTATIC_OBJECT_MANAGER.  With that
193      * #define, the application must create the ACE_Object_Manager.
194      * The recommended way is to call <ACE::init> at the start of
195      * the program, and call <ACE::fini> at the end.  Alternatively,
196      * the application could explicity construct an
197      * ACE_Object_Manager.
198      */

204       /**
205        * Explicitly initialize (construct the singleton instance of) the
206        * ACE_Object_Manager.  Returns 0 on success, -1 on failure, and 1
207        * if it had already been called.
208        */
209       virtual int init ();
210     
211       /**
212        * Explicitly destroy the singleton instance of the
213        * ACE_Object_Manager.  Returns 0 on success, -1 on failure, and 1
214        * if it had already been called.
215        */
216       virtual int fini ();
217     
218       /**
219        * Returns 1 before the ACE_Object_Manager has been constructed.
220        * This flag can be used to determine if the program is constructing
221        * static objects.  If no static object spawns any threads, the
222        * program will be single-threaded when this flag returns 1.  (Note
223        * that the program still might construct some static objects when
224        * this flag returns 0, if ACE_HAS_NONSTATIC_OBJECT_MANAGER is not
225        * defined.)
226        */
227       static int starting_up ();
228     
229       /**
230        * Returns 1 after the ACE_Object_Manager has been destroyed.  This
231        * flag can be used to determine if the program is in the midst of
232        * destroying static objects.  (Note that the program might destroy
233        * some static objects before this flag can return 1, if
234        * ACE_HAS_NONSTATIC_OBJECT_MANAGER is not defined.)
235        */
236       static int shutting_down ();
237     
238       /**
239        * Register an ACE_Cleanup object for cleanup at process
240        * termination.  The object is deleted via the
241        * <ace_cleanup_destroyer>.  If you need more flexibility, see the
242        * @c other at_exit method below.  For OS's that do not have
243        * processes, cleanup takes place at the end of <main>.  Returns 0
244        * on success.  On failure, returns -1 and sets errno to: EAGAIN if
245        * shutting down, ENOMEM if insufficient virtual memory, or EEXIST
246        * if the object (or array) had already been registered.
247        */
248       static int at_exit (ACE_Cleanup *object, void *param = nullptr, const char* name = nullptr);
249     
250     #if defined (ACE_HAS_TSS_EMULATION)
251       static int init_tss ();
252       int init_tss_i ();
253     #endif

255       /**
256        * Register an object (or array) for cleanup at process termination.
257        * "cleanup_hook" points to a (global, or static member) function
258        * that is called for the object or array when it to be destroyed.
259        * It may perform any necessary cleanup specific for that object or
260        * its class.  "param" is passed as the second parameter to the
261        * @a cleanup_hook function; the first parameter is the object (or
262        * array) to be destroyed.  @a cleanup_hook, for example, may delete
263        * the object (or array).  For OS's that do not have processes, this
264        * function is the same as <at_thread_exit>.  Returns 0 on success.
265        * On failure, returns -1 and sets errno to: EAGAIN if shutting
266        * down, ENOMEM if insufficient virtual memory, or EEXIST if the
267        * object (or array) had already been registered.
268        */
269       static int at_exit (void *object,
270                           ACE_CLEANUP_FUNC cleanup_hook,
271                           void *param,
272                           const char* name = 0);
273     
274       static int remove_at_exit (void *object);
