142     ACE_INLINE int
143     ACE_OS::putmsg (ACE_HANDLE handle, const struct strbuf *ctl,
144                     const struct strbuf *data, int flags)
145     {
146       ACE_OS_TRACE ("ACE_OS::putmsg");
147     #if defined (ACE_HAS_STREAM_PIPES)
148       return ::putmsg (handle, (ACE_STRBUF_TYPE) ctl, (ACE_STRBUF_TYPE) data, flags);
149     #else
150       ACE_UNUSED_ARG (flags);
151       ssize_t result;
152       if (ctl == 0 && data == 0)
153         {
154           errno = EINVAL;
155           return 0;
156         }
157       // Handle the two easy cases.
158       else if (ctl != 0)
159         {
160           result = ACE_OS::write (handle, ctl->buf, static_cast<size_t> (ctl->len));
161           return static_cast<int> (result);
162         }
163       else if (data != 0)
164         {
165           result = ACE_OS::write (handle, data->buf, static_cast<size_t> (data->len));
166           return static_cast<int> (result);
167         }
168       else
169         {
170           // This is the hard case.
171           char *buf;
172           const auto alloc {static_cast<unsigned int> (ctl->len) + static_cast<unsigned int> (data->len)};
173     #if defined (ACE_HAS_ALLOC_HOOKS)
174           ACE_ALLOCATOR_RETURN (buf, static_cast<char*> (ACE_Allocator::instance ()->malloc (alloc)), -1);
175     #else
176           ACE_NEW_RETURN (buf, char [alloc], -1);
177     #endif /* ACE_HAS_ALLOC_HOOKS */
178           ACE_OS::memcpy (buf, ctl->buf, static_cast<size_t> (ctl->len));
179           ACE_OS::memcpy (buf + ctl->len, data->buf, static_cast<size_t> (data->len));
180           result = ACE_OS::write (handle, buf, alloc);
181     #if defined (ACE_HAS_ALLOC_HOOKS)
182           ACE_Allocator::instance()->free(buf);
183     #else
184           delete [] buf;
185     #endif /* ACE_HAS_ALLOC_HOOKS */
186     
187           return static_cast<int> (result);
188         }
189     #endif /* ACE_HAS_STREAM_PIPES */
190     }
