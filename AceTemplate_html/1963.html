283     ACE_OS::cond_init (ACE_cond_t *cv,
284                        ACE_condattr_t &attributes,
285                        const char *name,
286                        void *arg)
287     {
288       // ACE_OS_TRACE ("ACE_OS::cond_init");
289       ACE_UNUSED_ARG (name);
290       ACE_UNUSED_ARG (arg);
291     # if defined (ACE_HAS_THREADS)
292     #   if defined (ACE_HAS_PTHREADS)
293       int result = -1;

295     #     if defined (ACE_PTHREAD_COND_T_INITIALIZE)
296       /* VxWorks 6.x API reference states:

315     # else
316       ACE_UNUSED_ARG (cv);
317       ACE_UNUSED_ARG (attributes);
318       ACE_UNUSED_ARG (name);
319       ACE_UNUSED_ARG (arg);
320       ACE_NOTSUP_RETURN (-1);
321     # endif /* ACE_HAS_THREADS */

384     ACE_INLINE int
385     ACE_OS::cond_timedwait (ACE_cond_t *cv,
386                             ACE_mutex_t *external_mutex,
387                             ACE_Time_Value *timeout)
388     {
389       ACE_OS_TRACE ("ACE_OS::cond_timedwait");
390     # if defined (ACE_HAS_THREADS)
391       int result = 0;
392       timespec_t ts;
393     
394       if (timeout != 0)
395         ts = *timeout; // Calls ACE_Time_Value::operator timespec_t().
396     
397     #   if defined (ACE_HAS_PTHREADS)
398     
399       ACE_OSCALL (ACE_ADAPT_RETVAL (timeout == 0
400                                     ? pthread_cond_wait (cv, external_mutex)
401                                     : pthread_cond_timedwait (cv, external_mutex,
402                                                                 (ACE_TIMESPEC_PTR) &ts),
403                                     result),
404                   int, result);
405       // We need to adjust this to make the POSIX return
406       // values consistent.  EAGAIN is from Pthreads DRAFT4
407       if (result == -1 &&
408           (errno == ETIMEDOUT || errno == EAGAIN))
409         errno = ETIME;
410     
411     #   elif defined (ACE_HAS_WTHREADS) && defined (ACE_HAS_WTHREADS_CONDITION_VARIABLE)
412       int msec_timeout = 0;
413       if (timeout != 0)
414         {
415           ACE_Time_Value relative_time = timeout->to_relative_time ();
416           // Watchout for situations where a context switch has caused the
417           // current time to be > the timeout.
418           if (relative_time > ACE_Time_Value::zero)
419             msec_timeout = relative_time.msec ();
420         }
421     
422       ACE_OSCALL (ACE_ADAPT_RETVAL (::SleepConditionVariableCS (cv, &external_mutex->thr_mutex_, msec_timeout),
423                                     result),
424                   int, result);
425     
426       return result;
427     #   endif /* ACE_HAS_PTHREADS */
428       if (timeout != 0)
429         timeout->set (ts); // Update the time value before returning.
430     
431       return result;
432     # else
433       ACE_UNUSED_ARG (cv);
434       ACE_UNUSED_ARG (external_mutex);
435       ACE_UNUSED_ARG (timeout);
436       ACE_NOTSUP_RETURN (-1);
437     # endif /* ACE_HAS_THREADS */
438     }
439     #endif /* !ACE_LACKS_COND_T */

551     // This method is used to prepare the recursive mutex for releasing
552     // when waiting on a condition variable. If the platform doesn't have
553     // native recursive mutex and condition variable support, then ACE needs
554     // to save the recursion state around the wait and also ensure that the
555     // wait and lock release are atomic. recursive_mutex_cond_relock()
556     // is the inverse of this method.
557     ACE_INLINE int
558     ACE_OS::recursive_mutex_cond_unlock (ACE_recursive_thread_mutex_t *m,
559                                          ACE_recursive_mutex_state &state)
560     {
561     #if defined (ACE_HAS_THREADS)
562       ACE_OS_TRACE ("ACE_OS::recursive_mutex_cond_unlock");
563     #  if defined (ACE_HAS_RECURSIVE_MUTEXES)
564       // Windows need special handling since it has recursive mutexes, but
565       // does not integrate them into a condition variable.
566     #    if defined (ACE_WIN32)
567       // For Windows, the OS takes care of the mutex and its recursion. We just
568       // need to release the lock one fewer times than this thread has acquired
569       // it. Remember how many times, and reacquire it that many more times when
570       // the condition is signaled.
571     
572       // We're using undocumented fields in the CRITICAL_SECTION structure
573       // and they've been known to change across Windows variants and versions./
574       // So be careful if you need to change these - there may be other
575       // Windows variants that depend on existing values and limits.
576     
577       state.relock_count_ = 0;
578       while (m->LockCount > 0 && m->RecursionCount > 1)
579         {
580           // This may fail if the current thread doesn't own the mutex. If it
581           // does fail, it'll be on the first try, so don't worry about resetting
582           // the state.
583           if (ACE_OS::recursive_mutex_unlock (m) == -1)
584             return -1;
585           ++state.relock_count_;
586         }
587     #    else /* not ACE_WIN32 */
588         // prevent warnings for unused variables
589         ACE_UNUSED_ARG (state);
590         ACE_UNUSED_ARG (m);
591     #    endif /* ACE_WIN32 */
592       return 0;
593     #  else /* ACE_HAS_RECURSIVE_MUTEXES */
594       // For platforms without recursive mutexes, we obtain the nesting mutex
595       // to gain control over the mutex internals. Then set the internals to say
596       // the mutex is available. If there are waiters, signal the condition
597       // to notify them (this is mostly like the recursive_mutex_unlock() method).
598       // Then, return with the nesting mutex still held. The condition wait
599       // will release it atomically, allowing mutex waiters to continue.
600       // Note that this arrangement relies on the fact that on return from
601       // the condition wait, this thread will again own the nesting mutex
602       // and can either set the mutex internals directly or get in line for
603       // the mutex... this part is handled in recursive_mutex_cond_relock().
604       if (ACE_OS::thread_mutex_lock (&m->nesting_mutex_) == -1)
605         return -1;
606     
607     #    if !defined (ACE_NDEBUG)
608       if (m->nesting_level_ == 0
609           || ACE_OS::thr_equal (ACE_OS::thr_self (), m->owner_id_) == 0)
610         {
611           ACE_OS::thread_mutex_unlock (&m->nesting_mutex_);
612           errno = EINVAL;
613           return -1;
614         }
615     #    endif /* ACE_NDEBUG */
616     
617       // To make error recovery a bit easier, signal the condition now. Any
618       // waiter won't regain control until the mutex is released, which won't
619       // be until the caller returns and does the wait on the condition.
620       if (ACE_OS::cond_signal (&m->lock_available_) == -1)
621         {
622           // Save/restore errno.
623           ACE_Errno_Guard error (errno);
624           ACE_OS::thread_mutex_unlock (&m->nesting_mutex_);
625           return -1;
626         }
627     
628       // Ok, the nesting_mutex_ lock is still held, the condition has been
629       // signaled... reset the nesting info and return _WITH_ the lock
630       // held. The lock will be released when the condition waits, in the
631       // caller.
632       state.nesting_level_ = m->nesting_level_;
633       state.owner_id_ = m->owner_id_;
634       m->nesting_level_ = 0;
635       m->owner_id_ = ACE_OS::NULL_thread;
636       return 0;
637     #  endif /* ACE_HAS_RECURSIVE_MUTEXES */
638     #else
639       ACE_UNUSED_ARG (m);
640       ACE_UNUSED_ARG (state);
641       ACE_NOTSUP_RETURN (-1);
642     #endif /* ACE_HAS_THREADS */
643     }

646     // This method is called after waiting on a condition variable when a
647     // recursive mutex must be reacquired. If the platform doesn't natively
648     // integrate recursive mutexes and condition variables, it's taken care
649     // of here (inverse of ACE_OS::recursive_mutex_cond_unlock).
650     ACE_INLINE void
651     ACE_OS::recursive_mutex_cond_relock (ACE_recursive_thread_mutex_t *m,
652                                          ACE_recursive_mutex_state &state)
653     {
654     #if defined (ACE_HAS_THREADS)
655       ACE_OS_TRACE ("ACE_OS::recursive_mutex_cond_relock");
656     #  if defined (ACE_HAS_RECURSIVE_MUTEXES)
657       // Windows need special handling since it has recursive mutexes, but
658       // does not integrate them into a condition variable.
659       // On entry, the OS has already reacquired the lock for us. Just
660       // reacquire it the proper number of times so the recursion is the same as
661       // before waiting on the condition.
662     #    if defined (ACE_WIN32)
663       while (state.relock_count_ > 0)
664         {
665           ACE_OS::recursive_mutex_lock (m);
666           --state.relock_count_;
667         }
668       return;
669     #    else /* not ACE_WIN32 */
670         // prevent warnings for unused variables
671         ACE_UNUSED_ARG (state);
672         ACE_UNUSED_ARG (m);
673     
674     #    endif /* ACE_WIN32 */
675     #  else
676       // Without recursive mutex support, it's somewhat trickier. On entry,
677       // the current thread holds the nesting_mutex_, but another thread may
678       // still be holding the ACE_recursive_mutex_t. If so, mimic the code
679       // in ACE_OS::recursive_mutex_lock that waits to acquire the mutex.
680       // After acquiring it, restore the nesting counts and release the
681       // nesting mutex. This will restore the conditions to what they were
682       // before calling ACE_OS::recursive_mutex_cond_unlock().
683       while (m->nesting_level_ > 0)
684         ACE_OS::cond_wait (&m->lock_available_, &m->nesting_mutex_);
685     
686       // At this point, we still have nesting_mutex_ and the mutex is free.
687       m->nesting_level_ = state.nesting_level_;
688       m->owner_id_ = state.owner_id_;
689       ACE_OS::thread_mutex_unlock (&m->nesting_mutex_);
690       return;
691     #  endif /* ACE_HAS_RECURSIVE_MUTEXES */
692     #else
693       ACE_UNUSED_ARG (m);
694       ACE_UNUSED_ARG (state);
695       return;
696     #endif /* ACE_HAS_THREADS */
697     }

754     ACE_INLINE int
755     ACE_OS::recursive_mutex_lock (ACE_recursive_thread_mutex_t *m)
756     {
757     #if defined (ACE_HAS_THREADS)
758     #if defined (ACE_HAS_RECURSIVE_MUTEXES)
759       return ACE_OS::thread_mutex_lock (m);
760     #else
761       ACE_thread_t const t_id = ACE_OS::thr_self ();
762       int result = 0;
763     
764       // Acquire the guard.
765       if (ACE_OS::thread_mutex_lock (&m->nesting_mutex_) == -1)
766         result = -1;
767       else
768       {
769         // If there's no contention, just grab the lock immediately
770         // (since this is the common case we'll optimize for it).
771         if (m->nesting_level_ == 0)
772           m->owner_id_ = t_id;
773           // If we already own the lock, then increment the nesting level
774           // and return.
775         else if (ACE_OS::thr_equal (t_id, m->owner_id_) == 0)
776         {
777               // Wait until the nesting level has dropped to zero, at
778               // which point we can acquire the lock.
779           while (m->nesting_level_ > 0)
780             ACE_OS::cond_wait (&m->lock_available_,
781                                 &m->nesting_mutex_);
782     
783               // At this point the nesting_mutex_ is held...
784           m->owner_id_ = t_id;
785         }
786     
787         // At this point, we can safely increment the nesting_level_ no
788         // matter how we got here!
789         ++m->nesting_level_;
790       }
791     
792       {
793         // Save/restore errno.
794         ACE_Errno_Guard error (errno);
795         ACE_OS::thread_mutex_unlock (&m->nesting_mutex_);
796       }
797       return result;
798     #endif /* ACE_HAS_RECURSIVE_MUTEXES */
799     #else
800       ACE_UNUSED_ARG (m);
801       ACE_NOTSUP_RETURN (-1);
802     #endif /* ACE_HAS_THREADS */
803     }

805     ACE_INLINE int
806     ACE_OS::recursive_mutex_lock (ACE_recursive_thread_mutex_t *m,
807                                   const ACE_Time_Value &timeout)
808     {
809     #if defined (ACE_HAS_THREADS)
810     #if defined (ACE_HAS_RECURSIVE_MUTEXES)
811       return ACE_OS::thread_mutex_lock (m, timeout);
812     #else
813       ACE_thread_t t_id = ACE_OS::thr_self ();
814       int result = 0;
815     
816       // Try to acquire the guard.
817       if (ACE_OS::thread_mutex_lock (&m->nesting_mutex_, timeout) == -1)
818         result = -1;
819       else
820         {
821           // If there's no contention, just grab the lock immediately
822           // (since this is the common case we'll optimize for it).
823           if (m->nesting_level_ == 0)
824             m->owner_id_ = t_id;
825           // If we already own the lock, then increment the nesting level
826           // and return.
827           else if (ACE_OS::thr_equal (t_id, m->owner_id_) == 0)
828             {
829               // Wait until the nesting level has dropped to zero, at
830               // which point we can acquire the lock.
831               while (m->nesting_level_ > 0)
832                 {
833                   result = ACE_OS::cond_timedwait (&m->lock_available_,
834                                                    &m->nesting_mutex_,
835                                                    const_cast <ACE_Time_Value *> (&timeout));
836     
837                   // The mutex is reacquired even in the case of a timeout
838                   // release the mutex to prevent a deadlock
839                   if (result == -1)
840                     {
841                       // Save/restore errno.
842                       ACE_Errno_Guard error (errno);
843                       ACE_OS::thread_mutex_unlock (&m->nesting_mutex_);
844     
845                       return result;
846                     }
847                 }
848     
849               // At this point the nesting_mutex_ is held...
850               m->owner_id_ = t_id;
851             }
852     
853           // At this point, we can safely increment the nesting_level_ no
854           // matter how we got here!
855           m->nesting_level_++;
856     
857           // Save/restore errno.
858           ACE_Errno_Guard error (errno);
859           ACE_OS::thread_mutex_unlock (&m->nesting_mutex_);
860         }
861       return result;
862     #endif /* ACE_HAS_RECURSIVE_MUTEXES */
863     #else
864       ACE_UNUSED_ARG (m);
865       ACE_UNUSED_ARG (timeout);
866       ACE_NOTSUP_RETURN (-1);
867     #endif /* ACE_HAS_THREADS */
868     }

983     # if !defined (ACE_LACKS_RWLOCK_T)
984     #  if defined (ACE_HAS_PTHREADS_UNIX98_EXT)
985       int result;
986       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (pthread_rwlock_rdlock (rw),

988                          int);

1015         rw->ref_count_++;

1017         ACE_OS::mutex_unlock (&rw->lock_);
1018     #   if defined (ACE_HAS_PTHREADS)
1019       ACE_PTHREAD_CLEANUP_POP (0);
1020     #   endif /* defined (ACE_HAS_PTHREADS) */

1085       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (::rw_trywrlock (rw), result), int);
1086     #  endif /* ACE_HAS_PTHREADS_UNIX98_EXT */
1087     # else /* NT, POSIX, and VxWorks don't support this natively. */
1088       int result = -1;
1089

1093     
1094           if (rw->ref_count_ != 0)
1095             {
1096               error = EBUSY;
1097               result = -1;
1098             }
1099           else

1102               result = 0;
1103             }
1104     
1105           ACE_OS::mutex_unlock (&rw->lock_);
1106         }
1107       return result;
1108     # endif /* ! ACE_LACKS_RWLOCK_T */

1111       ACE_NOTSUP_RETURN (-1);
1112     #endif /* ACE_HAS_THREADS */

1115     // Note that the caller of this method *must* already possess this
1116     // lock as a read lock.
1117     // return {-1 and no errno set means: error,
1118     //         -1 and errno==EBUSY set means: could not upgrade,
1119     //         0 means: upgraded successfully}

1121     ACE_INLINE int
1122     ACE_OS::rw_trywrlock_upgrade (ACE_rwlock_t *rw)
1123     {
1124       ACE_OS_TRACE ("ACE_OS::rw_trywrlock_upgrade");
1125     #if defined (ACE_HAS_THREADS)
1126     # if defined (ACE_HAS_PTHREADS_UNIX98_EXT) && !defined (ACE_LACKS_RWLOCK_T)
1127       int result;
1128       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (pthread_rwlock_trywrlock (rw),
1129                                            result),
1130                          int);
1131     # elif !defined (ACE_LACKS_RWLOCK_T)
1132       // Some native rwlocks
1133       ACE_UNUSED_ARG (rw);
1134       ACE_NOTSUP_RETURN (-1);
1135     # else /* NT, POSIX, and VxWorks don't support this natively. */
1136       // The ACE rwlock emulation does support upgrade . . .
1137       int result = 0;
1138     
1139     #   if defined (ACE_HAS_PTHREADS)
1140       ACE_PTHREAD_CLEANUP_PUSH (&rw->lock_);
1141     #   endif /* defined (ACE_HAS_PTHREADS) */
1142     
1143       if (ACE_OS::mutex_lock (&rw->lock_) == -1)
1144         return -1;
1145         // -1 means didn't get the mutex, error
1146       else if (rw->important_writer_)
1147         // an other reader upgrades already
1148         {
1149           result = -1;
1150           errno = EBUSY;
1151         }
1152       else
1153         {
1154           while (rw->ref_count_ > 1) // wait until only I am left
1155             {
1156               rw->num_waiting_writers_++; // prohibit any more readers
1157               rw->important_writer_ = true;
1158     
1159               if (ACE_OS::cond_wait (&rw->waiting_important_writer_, &rw->lock_) == -1)
1160                 {
1161                   result = -1;
1162                   // we know that we have the lock again, we have this guarantee,
1163                   // but something went wrong
1164                 }
1165               rw->important_writer_ = false;
1166               rw->num_waiting_writers_--;
1167             }
1168           if (result == 0)
1169             {
1170               // nothing bad happend
1171               rw->ref_count_ = -1;
1172               // now I am a writer
1173               // everything is O.K.
1174             }
1175         }
1176     
1177       ACE_OS::mutex_unlock (&rw->lock_);
1178     
1179     #   if defined (ACE_HAS_PTHREADS)
1180       ACE_PTHREAD_CLEANUP_POP (0);
1181     #   endif /* defined (ACE_HAS_PTHREADS) */
1182     
1183       return result;
1184     
1185     # endif /* ! ACE_LACKS_RWLOCK_T */
1186     #else
1187       ACE_UNUSED_ARG (rw);
1188       ACE_NOTSUP_RETURN (-1);
1189     #endif /* ACE_HAS_THREADS */
1190     }

1193     ACE_OS::rw_unlock (ACE_rwlock_t *rw)
1194     {
1195       ACE_OS_TRACE ("ACE_OS::rw_unlock");
1196     #if defined (ACE_HAS_THREADS)
1197     # if !defined (ACE_LACKS_RWLOCK_T)
1198     #  if defined (ACE_HAS_PTHREADS_UNIX98_EXT)
1199       int result;
1200       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (pthread_rwlock_unlock (rw),

1202                          int);

1205       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (::rw_unlock (rw), result), int);

1208       if (ACE_OS::mutex_lock (&rw->lock_) == -1)

1211       if (rw->ref_count_ > 0) // Releasing a reader.
1212         rw->ref_count_--;
1213       else if (rw->ref_count_ == -1) // Releasing a writer.
1214         rw->ref_count_ = 0;
1215       else
1216         {
1217           (void) ACE_OS::mutex_unlock (&rw->lock_);
1218           return -1; // @@ ACE_ASSERT (!"count should not be 0!\n");
1219         }
1220     
1221       int result = 0;
1222       ACE_Errno_Guard error (errno);
1223     
1224       if (rw->important_writer_ && rw->ref_count_ == 1)
1225         // only the reader requesting to upgrade its lock is left over.
1226         {
1227           result = ACE_OS::cond_signal (&rw->waiting_important_writer_);
1228           error = errno;
1229         }
1230       else if (rw->num_waiting_writers_ > 0 && rw->ref_count_ == 0)
1231         // give preference to writers over readers...
1232         {
1233           result = ACE_OS::cond_signal (&rw->waiting_writers_);
1234           error =  errno;
1235         }
1236       else if (rw->num_waiting_readers_ > 0 && rw->num_waiting_writers_ == 0)
1237         {
1238           result = ACE_OS::cond_broadcast (&rw->waiting_readers_);
1239           error = errno;
1240         }
1241     
1242       (void) ACE_OS::mutex_unlock (&rw->lock_);
1243       return result;
1244     # endif /* ! ace_lacks_rwlock_t */
1245     #else
1246       ACE_UNUSED_ARG (rw);
1247       ACE_NOTSUP_RETURN (-1);
1248     #endif /* ace_has_threads */
1249     }

1252     ACE_OS::rw_wrlock (ACE_rwlock_t *rw)
1253     {
1254       ACE_OS_TRACE ("ACE_OS::rw_wrlock");
1255     #if defined (ACE_HAS_THREADS)
1256     # if !defined (ACE_LACKS_RWLOCK_T)
1257     #  if defined (ACE_HAS_PTHREADS_UNIX98_EXT)
1258       int result;
1259       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (pthread_rwlock_wrlock (rw),
1260                                            result),
1261                          int);
1262     #  else /* ACE_HAS_PTHREADS_UNIX98_EXT */
1263       int result;
1264       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (::rw_wrlock (rw), result), int);
1265     #  endif /* ACE_HAS_PTHREADS_UNIX98_EXT */
1266     # else /* NT, POSIX, and VxWorks don't support this natively. */
1267     #   if defined (ACE_HAS_PTHREADS)
1268       ACE_PTHREAD_CLEANUP_PUSH (&rw->lock_);
1269     #   endif /* defined (ACE_HAS_PTHREADS) */
1270       int result = 0;
1271     
1272       if (ACE_OS::mutex_lock (&rw->lock_) == -1)
1273         result = -1; // -1 means didn't get the mutex.
1274       else
1275         {
1276           while (rw->ref_count_ != 0)
1277             {
1278               rw->num_waiting_writers_++;
1279     
1280               if (ACE_OS::cond_wait (&rw->waiting_writers_, &rw->lock_) == -1)
1281                 {
1282                   result = -2; // -2 means we need to release the mutex.
1283                   break;
1284                 }
1285     
1286               rw->num_waiting_writers_--;
1287             }
1288         }
1289       if (result == 0)
1290         rw->ref_count_ = -1;
1291       if (result != -1)
1292         ACE_OS::mutex_unlock (&rw->lock_);
1293     #   if defined (ACE_HAS_PTHREADS)
1294       ACE_PTHREAD_CLEANUP_POP (0);
1295     #   endif /* defined (ACE_HAS_PTHREADS) */
1296       return 0;
1297     # endif /* ! ACE_LACKS_RWLOCK_T */
1298     #else
1299       ACE_UNUSED_ARG (rw);
1300       ACE_NOTSUP_RETURN (-1);
1301     #endif /* ACE_HAS_THREADS */
1302     }

1847     ACE_INLINE int
1848     ACE_OS::sema_post (ACE_sema_t *s)
1849     {
1850       ACE_OS_TRACE ("ACE_OS::sema_post");
1851     # if defined (ACE_HAS_POSIX_SEM)
1852     #   if defined (ACE_HAS_POSIX_SEM_TIMEOUT) || defined (ACE_DISABLE_POSIX_SEM_TIMEOUT_EMULATION)
1853       return ::sem_post (s->sema_);
1854     #   else
1855       int result = -1;
1856     
1857       if (ACE_OS::mutex_lock (&s->lock_) == 0)
1858         {
1859           if (::sem_post (s->sema_) == 0)
1860             result = ACE_OS::cond_signal (&s->count_nonzero_);
1861     
1862           ACE_OS::mutex_unlock (&s->lock_);
1863         }
1864       return result;
1865     #   endif /* ACE_HAS_POSIX_SEM_TIMEOUT || ACE_DISABLE_POSIX_SEM_TIMEOUT_EMULATION */
1866     # elif defined (ACE_USES_FIFO_SEM)
1867       char    c = 1;
1868       if (ACE_OS::write (s->fd_[1], &c, sizeof (char)) == sizeof (char))
1869         return 0;
1870       return -1;
1871     # elif defined (ACE_HAS_THREADS)
1872     #   if defined (ACE_HAS_PTHREADS)
1873       int result = -1;
1874     
1875       if (ACE_OS::mutex_lock (&s->lock_) == 0)
1876         {
1877           // Always allow a waiter to continue if there is one.
1878           if (s->waiters_ > 0)
1879             result = ACE_OS::cond_signal (&s->count_nonzero_);
1880           else
1881             result = 0;
1882     
1883           s->count_++;
1884           ACE_OS::mutex_unlock (&s->lock_);
1885         }
1886       return result;
1887     #   elif defined (ACE_HAS_WTHREADS)
1888       ACE_WIN32CALL_RETURN (ACE_ADAPT_RETVAL (::ReleaseSemaphore (*s, 1, 0),
1889                                               ace_result_),
1890                             int, -1);
1891     #   elif defined (ACE_VXWORKS)
1892       return ::semGive (s->sema_);
1893     #   elif defined (ACE_INTEGRITY)
1894       return ::ReleaseSemaphore (*s) == Success ? 0 : -1;
1895     #   endif /* ACE_HAS_PTHREADS */
1896     # else
1897       ACE_UNUSED_ARG (s);
1898       ACE_NOTSUP_RETURN (-1);
1899     # endif /* ACE_HAS_POSIX_SEM */
1900     }

1925     ACE_INLINE int
1926     ACE_OS::sema_trywait (ACE_sema_t *s)
1927     {
1928       ACE_OS_TRACE ("ACE_OS::sema_trywait");
1929     # if defined (ACE_HAS_POSIX_SEM)
1930       // POSIX semaphores set errno to EAGAIN if trywait fails
1931       return ::sem_trywait (s->sema_);
1932     # elif defined (ACE_USES_FIFO_SEM)
1933       char  c;
1934       int     rc, flags;
1935     
1936       /* turn on nonblocking for s->fd_[0] */
1937       if ((flags = ACE_OS::fcntl (s->fd_[0], F_GETFL, 0)) < 0)
1938         return -1;
1939       flags |= O_NONBLOCK;
1940       if (ACE_OS::fcntl (s->fd_[0], F_SETFL, flags) < 0)
1941         return -1;
1942     
1943       // read sets errno to EAGAIN if no input
1944       rc = ACE_OS::read (s->fd_[0], &c, sizeof (char));
1945     
1946       /* turn off nonblocking for fd_[0] */
1947       if ((flags = ACE_OS::fcntl (s->fd_[0], F_GETFL, 0)) >= 0)
1948       {
1949         flags &= ~O_NONBLOCK;
1950         ACE_OS::fcntl (s->fd_[0], F_SETFL, flags);
1951       }
1952     
1953       return rc == 1 ? 0 : (-1);
1954     # elif defined (ACE_HAS_THREADS)
1955     #   if defined (ACE_HAS_PTHREADS)
1956       int result = -1;
1957     
1958       if (ACE_OS::mutex_lock (&s->lock_) == 0)
1959         {
1960           if (s->count_ > 0)
1961             {
1962               --s->count_;
1963               result = 0;
1964             }
1965           else
1966             errno = EBUSY;
1967     
1968           ACE_OS::mutex_unlock (&s->lock_);
1969         }
1970       return result;
1971     #   elif defined (ACE_HAS_WTHREADS)
1972       DWORD result = ::WaitForSingleObject (*s, 0);
1973     
1974       if (result == WAIT_OBJECT_0)
1975         return 0;
1976       else
1977         {
1978           if (result == WAIT_TIMEOUT)
1979             errno = EBUSY;
1980           else
1981             ACE_OS::set_errno_to_last_error ();
1982           // This is a hack, we need to find an appropriate mapping...
1983           return -1;
1984         }
1985     #   elif defined (ACE_VXWORKS)
1986       if (::semTake (s->sema_, NO_WAIT) == ERROR)
1987         if (errno == S_objLib_OBJ_UNAVAILABLE)
1988           {
1989             // couldn't get the semaphore
1990             errno = EBUSY;
1991             return -1;
1992           }
1993         else
1994           // error
1995           return -1;
1996       else
1997         // got the semaphore
1998         return 0;
1999     #   elif defined (ACE_INTEGRITY)
2000       return ::TryToObtainSemaphore (*s) == Success ? 0 : -1;
2001     #   endif /* ACE_HAS_PTHREADS */
2002     # else
2003       ACE_UNUSED_ARG (s);
2004       ACE_NOTSUP_RETURN (-1);
2005     # endif /* ACE_HAS_POSIX_SEM */
2006     }

2008     ACE_INLINE int
2009     ACE_OS::sema_wait (ACE_sema_t *s)
2010     {
2011       ACE_OS_TRACE ("ACE_OS::sema_wait");
2012     # if defined (ACE_HAS_POSIX_SEM)
2013       return ::sem_wait (s->sema_);
2014     # elif defined (ACE_USES_FIFO_SEM)
2015       char c;
2016       if (ACE_OS::read (s->fd_[0], &c, sizeof (char)) == 1)
2017         return 0;
2018       return -1;
2019     # elif defined (ACE_HAS_THREADS)
2020     #   if defined (ACE_HAS_PTHREADS)
2021       int result = 0;
2022     
2023       ACE_PTHREAD_CLEANUP_PUSH (&s->lock_);
2024     
2025       if (ACE_OS::mutex_lock (&s->lock_) != 0)
2026         result = -1;
2027       else
2028         {
2029           // Keep track of the number of waiters so that we can signal
2030           // them properly in <ACE_OS::sema_post>.
2031           s->waiters_++;
2032     
2033           // Wait until the semaphore count is > 0.
2034           while (s->count_ == 0)
2035             if (ACE_OS::cond_wait (&s->count_nonzero_,
2036                                    &s->lock_) == -1)
2037               {
2038                 result = -2; // -2 means that we need to release the mutex.
2039                 break;
2040               }
2041     
2042           --s->waiters_;
2043         }
2044     
2045       if (result == 0)
2046         --s->count_;
2047     
2048       if (result != -1)
2049         ACE_OS::mutex_unlock (&s->lock_);
2050       ACE_PTHREAD_CLEANUP_POP (0);
2051       return result < 0 ? -1 : result;
2052     
2053     #   elif defined (ACE_HAS_WTHREADS)
2054       switch (::WaitForSingleObject (*s, INFINITE))
2055         {
2056         case WAIT_OBJECT_0:
2057           return 0;
2058         default:
2059           // This is a hack, we need to find an appropriate mapping...
2060           ACE_OS::set_errno_to_last_error ();
2061           return -1;
2062         }
2063       /* NOTREACHED */
2064     #   elif defined (ACE_VXWORKS)
2065       return ::semTake (s->sema_, WAIT_FOREVER);
2066     #   elif defined (ACE_INTEGRITY)
2067       return ::WaitForSemaphore (*s) == Success ? 0 : -1;
2068     #   endif /* ACE_HAS_PTHREADS */
2069     # else
2070       ACE_UNUSED_ARG (s);
2071       ACE_NOTSUP_RETURN (-1);
2072     # endif /* ACE_HAS_POSIX_SEM */
2073     }

2075     ACE_INLINE int
2076     ACE_OS::sema_wait (ACE_sema_t *s, ACE_Time_Value &tv)
2077     {
2078       ACE_OS_TRACE ("ACE_OS::sema_wait");
2079     # if defined (ACE_HAS_POSIX_SEM)
2080     #   if defined (ACE_HAS_POSIX_SEM_TIMEOUT)
2081       int rc;
2082       timespec_t ts;
2083       ts = tv; // Calls ACE_Time_Value::operator timespec_t().
2084       ACE_OSCALL (::sem_timedwait (s->sema_, &ts), int, rc);
2085       if (rc == -1 && errno == ETIMEDOUT)
2086         errno = ETIME;  /* POSIX returns ETIMEDOUT but we need ETIME */
2087       return rc;
2088     #   elif !defined (ACE_DISABLE_POSIX_SEM_TIMEOUT_EMULATION)
2089       int result = 0;
2090       bool expired = false;
2091       ACE_Errno_Guard error (errno);
2092     
2093       ACE_PTHREAD_CLEANUP_PUSH (&s->lock_);
2094     
2095       if (ACE_OS::mutex_lock (&s->lock_) != 0)
2096         result = -2;
2097       else
2098         {
2099           bool finished = true;
2100           do
2101           {
2102             result = ACE_OS::sema_trywait (s);
2103             if (result == -1 && errno == EAGAIN)
2104               expired = (tv.to_relative_time () <= ACE_Time_Value::zero);
2105             else
2106               expired = false;
2107     
2108             finished = result != -1 || expired ||
2109                        (result == -1 && errno != EAGAIN);
2110             if (!finished)
2111               {
2112                 if (ACE_OS::cond_timedwait (&s->count_nonzero_,
2113                                             &s->lock_,
2114                                             &tv) == -1)
2115                   {
2116                     error = errno;
2117                     result = -1;
2118                     break;
2119                   }
2120               }
2121           } while (!finished);
2122     
2123           if (expired)
2124             error = ETIME;
2125         }
2126     
2127       if (result != -2)
2128         ACE_OS::mutex_unlock (&s->lock_);
2129       ACE_PTHREAD_CLEANUP_POP (0);
2130       return result < 0 ? -1 : result;
2131     #   else /* No native sem_timedwait(), and emulation disabled */
2132       ACE_UNUSED_ARG (s);
2133       ACE_UNUSED_ARG (tv);
2134       ACE_NOTSUP_RETURN (-1);
2135     #   endif /* ACE_HAS_POSIX_SEM_TIMEOUT */
2136     # elif defined (ACE_USES_FIFO_SEM)
2137       int rc;
2138       ACE_Time_Value timeout = tv.to_relative_time ();
2139     
2140       while (timeout > ACE_Time_Value::zero)
2141         {
2142           ACE_Handle_Set  fds_;
2143     
2144           fds_.set_bit (s->fd_[0]);
2145           if ((rc = ACE_OS::select (ACE_Handle_Set::MAXSIZE, fds_, 0, 0, timeout)) != 1)
2146             {
2147               if (rc == 0 || errno != EAGAIN)
2148               {
2149                 if (rc == 0)
2150                   errno = ETIME;
2151                 return -1;
2152               }
2153             }
2154     
2155           // try to read the signal *but* do *not* block
2156           if (rc == 1 && ACE_OS::sema_trywait (s) == 0)
2157             return 0;
2158     
2159           // we were woken for input but someone beat us to it
2160           // so we wait again if there is still time
2161           timeout = tv.to_relative_time ();
2162         }
2163     
2164       // make sure errno is set right
2165       errno = ETIME;
2166     
2167       return -1;
2168     # elif defined (ACE_HAS_THREADS)
2169     #   if defined (ACE_HAS_PTHREADS)
2170       int result = 0;
2171       ACE_Errno_Guard error (errno);
2172     
2173       ACE_PTHREAD_CLEANUP_PUSH (&s->lock_);
2174     
2175       if (ACE_OS::mutex_lock (&s->lock_) != 0)
2176         result = -1;
2177       else
2178         {
2179           // Keep track of the number of waiters so that we can signal
2180           // them properly in <ACE_OS::sema_post>.
2181           s->waiters_++;
2182     
2183           // Wait until the semaphore count is > 0 or until we time out.
2184           while (s->count_ == 0)
2185             if (ACE_OS::cond_timedwait (&s->count_nonzero_,
2186                                         &s->lock_,
2187                                         &tv) == -1)
2188               {
2189                 error = errno;
2190                 result = -2; // -2 means that we need to release the mutex.
2191                 break;
2192               }
2193     
2194           --s->waiters_;
2195         }
2196     
2197       if (result == 0)
2198         {
2199           --s->count_;
2200         }
2201     
2202       if (result != -1)
2203         ACE_OS::mutex_unlock (&s->lock_);
2204       ACE_PTHREAD_CLEANUP_POP (0);
2205       return result < 0 ? -1 : result;
2206     #   elif defined (ACE_HAS_WTHREADS)
2207       int msec_timeout;
2208     
2209       if (tv == ACE_Time_Value::zero)
2210         msec_timeout = 0; // Do a "poll."
2211       else
2212         {
2213           // Note that we must convert between absolute time (which is
2214           // passed as a parameter) and relative time (which is what
2215           // <WaitForSingleObjects> expects).
2216           ACE_Time_Value relative_time = tv.to_relative_time ();
2217     
2218           // Watchout for situations where a context switch has caused the
2219           // current time to be > the timeout.
2220           if (relative_time < ACE_Time_Value::zero)
2221             msec_timeout = 0;
2222           else
2223             msec_timeout = relative_time.msec ();
2224         }
2225     
2226       switch (::WaitForSingleObject (*s, msec_timeout))
2227         {
2228         case WAIT_OBJECT_0:
2229           tv = tv.now ();     // Update time to when acquired
2230           return 0;
2231         case WAIT_TIMEOUT:
2232           errno = ETIME;
2233           return -1;
2234         default:
2235           // This is a hack, we need to find an appropriate mapping...
2236           ACE_OS::set_errno_to_last_error ();
2237           return -1;
2238         }
2239       /* NOTREACHED */
2240     #   elif defined (ACE_VXWORKS)
2241       // Note that we must convert between absolute time (which is
2242       // passed as a parameter) and relative time (which is what
2243       // the system call expects).
2244       ACE_Time_Value relative_time = tv.to_relative_time ();
2245     
2246       _Vx_freq_t const ticks_per_sec = ::sysClkRateGet ();
2247     
2248       int ticks = relative_time.sec () * ticks_per_sec +
2249                   relative_time.usec () * ticks_per_sec / ACE_ONE_SECOND_IN_USECS;
2250       if (::semTake (s->sema_, ticks) == ERROR)
2251         {
2252           if (errno == S_objLib_OBJ_TIMEOUT)
2253             // Convert the VxWorks errno to one that's common for to ACE
2254             // platforms.
2255             errno = ETIME;
2256           else if (errno == S_objLib_OBJ_UNAVAILABLE)
2257             errno = EBUSY;
2258           return -1;
2259         }
2260       else
2261         {
2262           tv = tv.now ();  // Update to time acquired
2263           return 0;
2264         }
2265     #   elif defined (ACE_INTEGRITY)
2266       // INTEGRITY has kernel calls to wait for semaphore with timeout.
2267       // Since we are imitating INTEGRITY-178 behavior, we're not using them now.
2268       ACE_UNUSED_ARG (s);
2269       ACE_UNUSED_ARG (tv);
2270       ACE_NOTSUP_RETURN (-1);
2271     #   endif /* ACE_HAS_PTHREADS */
2272     # else
2273       ACE_UNUSED_ARG (s);
2274       ACE_UNUSED_ARG (tv);
2275       ACE_NOTSUP_RETURN (-1);
2276     # endif /* ACE_HAS_POSIX_SEM */
2277     }
