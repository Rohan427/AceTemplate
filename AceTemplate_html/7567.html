46     /**
47      * @class ACE_Service_Gestalt
48      *
49      * @brief Supplies common server operations for dynamic and static
50      * configuration of services.
51      *
52      * The Gestalt embodies the concept of configuration context. On one
53      * hand, it is a flat namespace, where names correspond to a Service
54      * Object instance. A Gestalt owns the Service Repository instance,
55      * which in turn owns the Service Object instances.
56      *
57      * Another aspect of a Gestalt is its responsibility for
58      * record-keeping and accounting for the meta-data, necessary for
59      * locating, removing or instantiating a service.
60      *
61      * A repository underlies an instance of a gestalt and its lifetime
62      * may or may not be bounded by the lifetime of the gestalt, that owns
63      * it. This feature is important for the derived classes and the
64      * Service Config in particular.
65      */

118       /**
119        * This is the primary entry point into the ACE_Service_Config (the
120        * constructor just handles simple initializations).  It parses
121        * arguments passed in from @a argc and @a argv parameters.  The
122        * arguments that are valid in a call to this method include:
123        *
124        * - '-b' Option to indicate that we should be a daemon. Note that when
125        *        this option is used, the process will be daemonized before the
126        *        service configuration file(s) are read. During daemonization,
127        *        (on POSIX systems) the current directory will be changed to "/"
128        *        so the caller should either fully specify the file names, or
129        *        execute a @c chroot() to the appropriate directory.
130        *        @sa ACE::daemonize().
131        * - '-d' Turn on debugging mode
132        * - '-f' Specifies a configuration file name other than the default
133        *        svc.conf. Can be specified multiple times to use multiple files.
134        *        If any configuration file is provided with this option then
135        *        the default svc.conf will be ignored.
136        * - '-k' Specifies the rendezvous point to use for the ACE distributed
137        *        logger.
138        * - '-y' Explicitly enables the use of static services. This flag
139        *        overrides the @a ignore_static_svcs parameter value.
140        * - '-n' Explicitly disables the use of static services. This flag
141        *        overrides the @a ignore_static_svcs parameter value.
142        * - '-p' Specifies a pathname which is used to store the process id.
143        * - '-s' Specifies a signal number other than SIGHUP to trigger reprocessing
144        *        of the configuration file(s). Ignored for platforms that do not
145        *        have POSIX signals, such as Windows.
146        * - '-S' Specifies a service directive string. Enclose the string in quotes
147        *        and escape any embedded quotes with a backslash. This option
148        *        specifies service directives without the need for a configuration
149        *        file. Can be specified multiple times.
150        *
151        * Note: Options '-f' and '-S' complement each other. Directives
152        * from files and from '-S' option are processed together in the
153        * following order. First, the default file "./svc.conf" is
154        * evaluated if not ignored, then all files are processed in the
155        * order they are specified in '-f' @a argv parameter. Finally, all
156        * '-S' directive strings are executed in the order the directives
157        * appear in @a argv parameter.
158        *
159        * If no files or directives are added via the '-f' and '-S'
160        * arguments, and the default file is not ignored, it will be
161        * evaluated whether it exists or not, possibly causing a failure
162        * return. If any other directives are added then the default file
163        * will be evaluated only if it exists.
164        *
165        * @param argc The number of commandline arguments.
166        * @param argv The array with commandline arguments
167        * @param logger_key   Indicates where to write the logging output,
168        *                     which is typically either a STREAM pipe or a
169        *                     socket address.
170        * @param ignore_static_svcs   If true then static services are not loaded,
171        *                             otherwise, they are loaded.
172        * @param ignore_default_svc_conf_file  If false then the @c ./svc.conf
173        *                                      configuration file will be ignored.
174        * @param ignore_debug_flag If false then the application is responsible
175        *                          for setting the @c ACE_Log_Msg::priority_mask
176        *                          appropriately.
177        *
178        * @retval -1   A configuration file is not found or cannot
179        *              be opened (errno is set accordingly).
180        * @retval  0   Success.
181        * @retval  >0  The number of directive errors encountered while processing
182        *              the service configuration file(s).
183        */
184       int open (int argc,
185                 ACE_TCHAR *argv[],
186                 const ACE_TCHAR *logger_key = 0,
187                 bool ignore_static_svcs = true,
188                 bool ignore_default_svc_conf_file = false,
189                 bool ignore_debug_flag = false);

472     /**
473      * @class ACE_Service_Type_Dynamic_Guard
474      *
475      * @brief A forward service declaration guard.
476      *
477      * Helps to resolve an issue with hybrid services, i.e. dynamic
478      * services, accompanied by static services in the same DLL.  Only
479      * automatic instances of this class are supposed to exist. Those are
480      * created during (dynamic) service initialization and serve to:
481      *
482      * (a) Ensure the service we are loading is ordered last in the
483      * repository, following any other services it may cause to register,
484      * as part of its own registration. This is a common case when
485      * loading dynamic services from DLLs - there are often static
486      * initializers, which register static services.
487      *
488      * (b) The SDG instance destructor detects if the dynamic service
489      * initialized successfully and "fixes-up" all the newly registered
490      * static services to hold a reference to the DLL, from which they
491      * have originated.
492      */
493     class ACE_Export ACE_Service_Type_Dynamic_Guard
494     {
495     public:
496       ACE_Service_Type_Dynamic_Guard (ACE_Service_Repository &r,
497                                       ACE_TCHAR const *name);
498     
499       ~ACE_Service_Type_Dynamic_Guard ();
500     
501     private:
502       ACE_Service_Repository & repo_;
503       size_t repo_begin_;
504       ACE_TCHAR const * const name_;
505     
506     #if defined (ACE_MT_SAFE) && (ACE_MT_SAFE != 0)
507     // FUZZ: disable check_for_ACE_Guard
508       ACE_Guard< ACE_Recursive_Thread_Mutex > repo_monitor_;
509     // FUZZ: enable check_for_ACE_Guard
