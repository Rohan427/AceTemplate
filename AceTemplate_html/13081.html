14     static void signalHandler (int sig)
15     {
16         const char* sigName = "unknown";
17         switch (sig)
18         {
19             case SIGINT:  sigName = "SIGINT";  break;
20             case SIGTERM: sigName = "SIGTERM"; break;
21             case SIGHUP:  sigName = "SIGHUP";  break;
22             case SIGSEGV: sigName = "SIGSEGV"; break;
23                 // Add more as needed
24         }
25     
26         ACE_DEBUG ((LM_INFO, ACE_TEXT ("Received signal %s (%d) - shutting down...\n"), sigName, sig));
27     
28         if (sig == SIGSEGV)
29         {
30             ACE_DEBUG ((LM_CRITICAL, ACE_TEXT ("Segmentation fault - dumping core if enabled\n")));
31         }
32     
33         Manager::DataProcessor::instance()->shutdown();
34         ACE_OS::exit (0);   // Or return from main if possible
35     }

39     float generateFloat (float min, float max)
40     {
41         double scaled = static_cast<double> (std::rand()) / RAND_MAX;
42         return min + static_cast<float> (scaled * (max - min + 1));
43     }

45     float generateInt (int min, int max)
46     {
47         double scaled = static_cast<double> (std::rand()) / RAND_MAX;
48         return min + static_cast<int> (scaled * (max - min + 1));
49     }
