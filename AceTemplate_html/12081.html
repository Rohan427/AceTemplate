139     bool Config::update()
140     {
141         struct stat fileInfo;
142     
143         if (!current)
144         {
145             // read file if exists
146     #ifdef DEBUG
147             std::cout << "Parse JSON file with comments" << std::endl;
148     #endif
149             if (stat (getFilePath().c_str(), &fileInfo) == 0)
150             {
151                 // st_mtime is the time of last modification of file content
152                 time_t mtime = fileInfo.st_mtime;
153     
154                 if ((getLastReadTime() == 0) || (mtime > getLastReadTime()))
155                 {
156                     if ((current = parser.read2 (getFilePath().c_str())))
157                     {
158                         
159                         getThreadSleepTime(); // 50ms tick
160                         getMaxThreads(); // The maximum number of threads allocated to the application
161                         getMaxObjects();
162                         getMaxFpuThreads();
163                         getLevel();
164                         getDataUpdateInterval();
165                         getMaxConThreads();
166                         getMaxProdThreads();
167     
168                         std::time (&timestamp);
169                     }
170                 } // if ((getLastReadTime() == 0) || (mtime > getLastReadTime()))
171                 else
172                 {
173                     std::cerr << getFilePath() << " not updated" << std::endl;
174                     errObj.err = parser.getJsonError().err;
175                     errObj.message = parser.getJsonError().message;
176                     errObj.varName = parser.getJsonError().varName;
177                     errObj.type = parser.getJsonError().type;
178                 } // END IF-ELSE: if ((getLastReadTime() == 0) || (mtime > getLastReadTime()))
179             }
180             else
181             {
182                 std::cerr << "Error getting file info " << getFilePath() << ": " << strerror (errno) << std::endl;
183                 errObj.err = errno;
184                 errObj.message = strerror (errno);
185                 errObj.varName = getFilePath();
186                 errObj.type = json::Json::TYPE_NULL;
187             } // END IF-ELSE: if (stat (getFilePath(), &fileInfo) != 0)
188     #ifdef DEBUG
189             std::cout << "File update complete: " << std::endl << std::endl;
190     #endif
191         }
192         // else don't read
193     
194         return (current);
195     }
