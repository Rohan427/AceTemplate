11     ACE_INLINE int
12     kill (pid_t pid, int signum)
13     {
14       ACE_OS_TRACE ("ACE_OS::kill");
15     #if defined (ACE_LACKS_KILL)
16       ACE_UNUSED_ARG (pid);
17       ACE_UNUSED_ARG (signum);
18       ACE_NOTSUP_RETURN (-1);
19     #elif defined (ACE_VXWORKS)
20       /*
21        * The VxWorks kill interface is not really POSIX
22        * since they use a task id in place of a pid type.
23        * This only becomes an issue when using the 64bit compiler
24        * as the TASK_ID is no longer defined as an int.
25        */
26       return ::kill ((ACE_VX_TASK_ID)pid, signum);
27     #else
28       return ::kill (pid, signum);
29     #endif /* ACE_LACKS_KILL */
30     }

32     ACE_INLINE int
33     pthread_sigmask (int how, const sigset_t *nsp, sigset_t *osp)
34     {
35     #if defined (ACE_HAS_PTHREADS) && !defined (ACE_LACKS_PTHREAD_SIGMASK)
36       int result;
37     # ifdef ACE_PTHREAD_SIGMASK_MACRO
38       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (ACE_PTHREAD_SIGMASK_MACRO (how, nsp, osp), result), int);
39     # elif defined (ACE_HAS_NONCONST_PTHREAD_SIGMASK)
40       sigset_t *ncnsp = const_cast<sigset_t *>(nsp);
41       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (::pthread_sigmask (how, ncnsp, osp), result), int);
42     # else
43       ACE_OSCALL_RETURN (ACE_ADAPT_RETVAL (::pthread_sigmask (how, nsp, osp), result), int);
44     # endif /* ACE_HAS_NONCONST__PTHREAD_SIGMASK */
45     #else /* !ACE_HAS_PTHREADS && !ACE_LACKS_PTHREAD_SIGMASK */
46       ACE_UNUSED_ARG (how);
47       ACE_UNUSED_ARG (nsp);
48       ACE_UNUSED_ARG (osp);
49       ACE_NOTSUP_RETURN (-1);
50     #endif /* ACE_HAS_PTHREADS && !ACE_LACKS_PTHREAD_SIGMASK */
51     }

53     ACE_INLINE int
54     sigaction (int signum, const ACE_SIGACTION *nsa, ACE_SIGACTION *osa)
55     {
56       ACE_OS_TRACE ("ACE_OS::sigaction");
57       if (signum == 0)
58         return 0;
59     #if defined (ACE_WIN32)
60       struct sigaction sa;
61     
62       if (osa == 0)
63         osa = &sa;
64     
65       if (nsa == 0)
66         {
67           osa->sa_handler = ::signal (signum, SIG_IGN);
68           ::signal (signum, osa->sa_handler);
69         }
70       else
71         osa->sa_handler = ::signal (signum, nsa->sa_handler);
72       return osa->sa_handler == SIG_ERR ? -1 : 0;
73     #elif defined (ACE_LACKS_SIGACTION) || (defined (ACE_INTEGRITY) && defined (ACE_LACKS_POSIX))
74       // INTEGRITY has the sigaction struct but the sigaction call may still be unavailable.
75       ACE_UNUSED_ARG (nsa);
76       ACE_UNUSED_ARG (osa);
77       ACE_NOTSUP_RETURN (-1);
78     #elif !defined (ACE_HAS_SIGACTION_CONSTP2)
79       return ::sigaction (signum, const_cast<ACE_SIGACTION*> (nsa), osa);
80     #else
81       return ::sigaction (signum, nsa, osa);
82     #endif /* ACE_WIN32 */
83     }

85     ACE_INLINE int
86     sigaddset (sigset_t *s, int signum)
87     {
88       ACE_OS_TRACE ("ACE_OS::sigaddset");
89     #if defined (ACE_LACKS_SIGADDSET)
90       ACE_UNUSED_ARG (s);
91       ACE_UNUSED_ARG (signum);
92       ACE_NOTSUP_RETURN (-1);
93     #elif defined (ACE_LACKS_SIGSET)
94       if (s == 0)
95         {
96           errno = EFAULT;
97           return -1;
98         }
99       else if (signum < 1 || signum >= ACE_NSIG)
100         {
101           errno = EINVAL;
102           return -1;                 // Invalid signum, return error
103         }
104       *s |= (1 << (signum - 1)) ;
105       return 0 ;
106     #else
107       return ace_sigaddset_helper (s, signum);
108     #endif
109     }

174     ACE_INLINE int
175     sigismember (sigset_t *s, int signum)
176     {
177     #if defined (ACE_LACKS_SIGISMEMBER)
178       ACE_UNUSED_ARG (s);
179       ACE_UNUSED_ARG (signum);
180       ACE_NOTSUP_RETURN (-1);
181     #elif defined (ACE_LACKS_SIGSET)
182       if (s == 0)
183         {
184           errno = EFAULT;
185           return -1;
186         }
187       else if (signum < 1 || signum >= ACE_NSIG)
188         {
189           errno = EINVAL;
190           return -1;                 // Invalid signum, return error
191         }
192       return ((*s & (1 << (signum - 1))) != 0) ;
193     #else
194     #  if defined (ACE_HAS_SIGISMEMBER_BUG)
195       if (signum < 1 || signum >= ACE_NSIG)
196         {
197           errno = EINVAL;
198           return -1;                 // Invalid signum, return error
199         }
200     #  endif /* ACE_HAS_SIGISMEMBER_BUG */
201       return ace_sigismember_helper (s, signum);
202     #endif
203     }

205     ACE_INLINE ACE_SignalHandler
206     signal (int signum, ACE_SignalHandler func)
207     {
208       if (signum == 0)
209         return 0;
210       else
211         {
212     #if (defined ACE_WIN32) || \
213         (!defined ACE_LACKS_UNIX_SIGNALS && !defined ACE_LACKS_SIGNAL)
214           return ::signal (signum, func);
215     #else
216           ACE_UNUSED_ARG (signum);
217           ACE_UNUSED_ARG (func);
218           ACE_NOTSUP_RETURN (0);     // Should return SIG_ERR but maybe not defined on all platforms
219     #endif /* defined (ACE_WIN32) || !defined (ACE_LACKS_UNIX_SIGNALS) */
220         }
221     }
