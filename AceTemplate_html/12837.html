14         void DataProcessor::initialize (size_t poolCapacity)
15         {
16             ACE_DEBUG ((LM_DEBUG, ACE_TEXT ("[%T][%M][TID:%t]: DataProcessor::initialize\n")));
17     
18             // Producer (long-running)
19             TaskConfig prodCfg;
20             prodCfg.numThreads = ::Config::getInstance().MAX_PROD_THREADS; // ::Config::getInstance().getMaxThreads() / 2;
21             prodCfg.basePriority = 30;
22             prodCfg.workloadType = WorkloadType::PRIMARY_WORKER;
23             prodCfg.useRealTime = true;
24             prodCfg.intervalMs = 0;
25             prodCfg.name = "DataProducer";
26             prodCfg.role = ROLE_PRODUCER;
27             prodCfg.producerIntervalMs = ::Config::getInstance().UPDATE_INTERVAL;  // 120000 is 2 minutes
28             prodCfg.tier = SchedulingTier::RealTimeAndAffinity;
29     
30             // Consumer (fast)
31             TaskConfig consCfg;
32             consCfg.numThreads = ::Config::getInstance().MAX_CON_THREADS;
33             consCfg.basePriority = 50;
34             consCfg.workloadType = WorkloadType::SIBLING_WORKER;
35             consCfg.useRealTime = true;
36             consCfg.intervalMs = 0;
37             consCfg.name = "DataConsumer";
38             consCfg.role = ROLE_CONSUMER;
39             consCfg.producerIntervalMs = 0;
40             consCfg.tier = SchedulingTier::RealTimeAndAffinity;
41     
42             ACE_DEBUG ((LM_DEBUG, ACE_TEXT ("[%T][%M][TID:%t]: DataProcessor::initialize tasks with %i objects in buffers\n"), poolCapacity));
43     
44             m_orchestrator->initialize (::Config::getInstance().getMaxThreads());
45     
46             m_orchestrator->createProducer (prodCfg);
47             m_orchestrator->createConsumer (consCfg);
48         }

50         void DataProcessor::acceptData (std::vector<TestData>* dataPtr)
51         {
52             ACE_DEBUG ((LM_DEBUG, ACE_TEXT ("[%T][%M][TID:%t]: DataProcessor::acceptData, with data size of %i\n"), dataPtr->size()));
53     
54             m_sharedDataPtr = dataPtr;
55     
56             m_orchestrator->runProducer ("DataProducer", dataPtr, 0);
57     
58             if (m_orchestrator->isDataValid())
59             {
60                 m_orchestrator->updateSharedDataPtr (dataPtr);
61                 m_orchestrator->runConsumer ("DataConsumer", dataPtr, 0);
62                 m_orchestrator->waitForConsumerCompletion ("DataConsumer", TaskRole::ROLE_CONSUMER);
63             }
64         }

69

77                 TaskOrchestrator<ObjectData, TestData>::instance().stopAll();
78             }
79             // Any other cleanup
80         }
